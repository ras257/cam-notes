\chapter{Minimising Convex Functions}
\section{Introduction to Minimisation}
In this course we will study problems of the form:
\begin{mini*}
  {\vec{x} \in \mathcal{X}}{f(\vec{x})}{}{}
  \addConstraint{h(\vec{x})}{= \vec{b}}
\end{mini*}
Without loss in generality, we can also assume that we are minimising functions as maximising $f(\vec{x})$ is equivalent to minimising $-f(\vec{x})$.

We may also have a \textit{inequality constraint}, such as $h(\vec{x}) \leq \vec{b}$ where we interpret this as $y_i \leq b_i$ for each of the components of $h(\vec{x}) = \vec{y}$ and $\vec{b}$.

The function $f: \R^{n} \to \R$ is known as the \textit{objective function}, $h(\vec{x}) = \vec{b}$ is known as a \textit{functional constraint}, and $x \in \mathcal{X}$ is known as a \textit{regional constraint}.

If $\vec{x}^{*}$ solves the problem, then we call it the \textit{optimal solution}, and for this value $f(\vec{x}^{*})$ is known as the \textit{optimal value/cost}.

The set:
\[
  \mathcal{X}(\vec{b}) = \{\vec{x} \in \R^{n}: h(\vec{x}) = \vec{b},\ \vec{x} \in \mathcal{X}\}
\]
is called the \textit{feasible set}.

We can in fact assume that we always have an equality constraint as $h(\vec{x}) \leq \vec{b}$ is the same as the constraint $h(\vec{x}) + \vec{s} = \vec{b},\ \vec{s} \geq 0$.
These additional variables $\vec{s}$ are called \textit{slack variables}.
\section{Convexity}
\subsection{Defining Convexity}
\begin{definition}[Convex Set]
  A set $S \subseteq \R^{n}$ is called \textit{convex} if for all $\vec{x}, \vec{y} \in S$ and for every $\lambda \in [0, 1]$, the point $\lambda \vec{x} + (1 - \lambda)\vec{y}$ is also in $S$.
\end{definition}
\begin{remark}[Intuition]
  A set is convex if for any two points, every point on the line segment joining these to points is contained within the set.
\end{remark}
\begin{definition}[Convex Function]
  A function $f: S \to \R$  is called a \textit{convex function} if $S$ is convex and for every $\vec{x}, \vec{y} \in S$ and $\lambda \in [0, 1]$ we have:
  \[
    f(\lambda \vec{x} + (1 - \lambda)\vec{y}) \leq \lambda f(\vec{x}) + (1 - \lambda) f(\vec{y})
  \]
\end{definition}
\begin{remark}[Intuition]
  A function is convex if the function value evaluated at a point on the line segment between any two points is less than the chord between those two points on the curve.
\end{remark}
\begin{remark}
  \begin{itemize}
    \item If $-f$ is convex, then we call $f$ \textit{concave}.
    \item If $f$ is linear, then this inequality always holds and so linear functions are always both convex and concave.
  \end{itemize}
\end{remark}
\subsection{Conditions for Convexity}
\begin{theorem}[First-order Condition for Convexity]
  A differentiable function $f: \R^{n} \to \R$ is convex if and only if for all $\vec{x}, \vec{y} \in \R^{n}$ we have: \label{firstOrderCond}
  \[
    f(\vec{y}) \geq f(\vec{x}) + (\vec{y} - \vec{x})^{\trans} \nabla f(\vec{x})
  \]
\end{theorem}
\begin{remark}
  This tells us that if $f$ is convex and we have a point $\vec{x}$ with $\nabla f(\vec{x}) = \vec{0}$, then $\vec{x}$ is the minimiser of $f$ as then $f(\vec{y}) \geq f(\vec{x})\ \forall \vec{y} \in \R^{n}$.
\end{remark}
\begin{proof}
  \begin{proofdirection}{Assume $f$ is convex}
    When $n = 1$, we have:
    \[
      f(x + t(y - x)) \leq (1 - t)f(x) + tf(y)\ \forall t \in [0, 1]
    \]
    This means that
    \[
      f(y) \geq f(x) + \frac{f(x + t(y - x)) - f(x)}{t}
    \]
    Taking $t \to 0$, since $f$ is differentiable:
    \[
      f(y) \geq f(x) + (y - x)f'(x)
    \]
    For the general case, we define:
    \[
      g: [0, 1] \to \R,\ t \mapsto f((1 - t)\vec{x} + t\vec{y})
    \]
    $g$ is convex because $f$ is a convex function so using the 1D result:
    \begin{align*}
      g(1) &\geq g(0) + (1 - 0)g'(0) \\
      \implies f(\vec{y}) &= f(\vec{x}) + 1 \cdot \at{\deriv{}{t}[(1 - t)\vec{x} + t\vec{y}]}{t = 0} [\nabla f((1 - 0)\vec{x} + 0 \cdot \vec{y})]^{\trans} \\
                          &= f(\vec{x}) + (\vec{y} - \vec{x}) [\nabla f(\vec{x})]^{\trans}
    \end{align*}
  \end{proofdirection}
  \begin{proofdirection}{Assume the inequality holds}
    Let $\vec{x}_t = (1 - t)\vec{x} + t\vec{y}$, the first order conditions tells us that:
    \begin{align*}
      f(\vec{x}) \geq f(\vec{x}_t) + \nabla f(\vec{x}_t)^{\trans} (\vec{x} - \vec{x}_t) \\
      f(\vec{y}) \geq f(\vec{x}_t) + \nabla f(\vec{x}_t)^{\trans} (\vec{y} - \vec{x}_t)
    \end{align*}
    Multiplying the first inequality by $(1 - t)$ and the second by $t$ and summing, we have that:
    \begin{align*}
      (1 - t)f(\vec{x}) + tf(\vec{y}) &\geq f(\vec{x}_t) \\
                                      &= f((1 - t)\vec{x} + t\vec{y})
    \end{align*}
  \end{proofdirection}
\end{proof}
\begin{definition}[Hessian]
  We define the \textit{hessian matrix} $\nabla^2 f(\vec{x})$ as:
  \[
    [\nabla^2 f(\vec{x})]_{i j} = \frac{\partial^2 f(\vec{x})}{\partial x_i \partial x_j}
  \]
\end{definition}
\begin{remark}[Recap]
  Recall from IA Differential Equations that:
  \[
    f(\vec{y}) = f(\vec{x}) + (\vec{y} - \vec{x})^{\trans} \nabla f(\vec{x}) + \frac{1}{2}(\vec{y} - \vec{x})^{\trans} [\nabla^2 f(\vec{x})] (\vec{y} - \vec{x}) + \cdots
  \]
  This is derived by applying the 1D Taylor series to $g(t) = f(\vec{x} + (\vec{y} - \vec{x})t)$.
\end{remark}
\begin{definition}[Positive Semi-definite]
  A symmetric matrix $A$ is \textit{positive semi-definite} if for any $\vec{x} \in \R^{n}$ we have:
  \[
    \vec{x}^{\trans} A \vec{x} \geq 0
  \]
  We then write $A \succeq 0$.
  Equivalently, all of the eigenvalues of $A$ are non-negative.
\end{definition}
\begin{theorem}[Second-order Condition for Convexity]
  A twice differentiable function $f: \R^{n} \to \R$ is \textit{convex} if and only if its hessian $\nabla^2 f(\vec{x}) \succeq 0$.
\end{theorem}
\begin{proof}
  Here will only prove one direction however the converse also holds.
\begin{proofdirection}{Assume $f$ is convex}
    For any $\vec{x}, \vec{y} \in \R^{n}$, by applying the Lagrange remainder to $g(t)$ as defined above, there exists $\vec{z} \in \R^{n}$ on the line joining $\vec{x}$ and $\vec{y}$ such that:
    \begin{align*}
      f(\vec{y}) &= f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x}) + \frac{1}{2}(\vec{y} - \vec{x})^{\trans} \nabla^2 f(\vec{z}) (\vec{y} - \vec{x}) \\
                 &\geq f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x})
    \end{align*}
    So by \cref{firstOrderCond}, $f$ is convex.
  \end{proofdirection}
\end{proof}
\section{Gradient Descent}
\subsection{Algorithm}
Observe that:
\[
  f(\vec{y}) \approx f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x})
\]
If $\vec{y} - \vec{x} = -\varepsilon\nabla f(\vec{x})$ then:
\begin{align*}
  f(\vec{y}) &\approx f(\vec{x}) + \nabla f(\vec{x})^{\trans}(-\varepsilon \nabla f(\vec{x})) \\
             &= f(\vec{x}) - \varepsilon \norm{\nabla f(\vec{x})}^2
\end{align*}
\subsubsection{Steps for Gradient Descent}
\begin{enumerate}
  \item Start from some $\vec{x}_0$ with $t = 0$.
  \item Repeat:
    \begin{enumerate}
      \item Find a descent direction $\vec{\nu}_t$ such as $-\nabla f(\vec{x}_t)$.
      \item Choose a step size $\eta_t$
      \item Update $\vec{x}_{t + 1} = \vec{x}_t + \eta_t \vec{\nu}_t$
    \end{enumerate}
  \item Stop when some stopping criteria are satisfied.
    For example $\nabla f(\vec{x}_t)$ is sufficiently close to $0$.
\end{enumerate}
There are however some potential issues with gradient descent:
\begin{itemize}
  \item The step size may need to be modified.
    For very curved functions, a large step size may just oscillate around the minimizer and for very flat functions, a small step size is quite inefficient.
\end{itemize}
\subsection{Smoothness and Strong Convexity}
\begin{definition}[$\beta$-smooth]
  A twice-differentiable function $f: \R^{n} \to \R$ is \textit{$\beta$-smooth} for $\beta > 0$ if $\nabla^2 f(\vec{x}) \preceq \beta I$ for all $\vec{x} \in \R^{n}$.
\end{definition}
\begin{definition}[$\alpha$-strong convexity]
  A twice-differentiable function $f: \R^{n} \to \R$ is \textit{$\alpha$-strongly convex} for $\alpha > 0$ if $\nabla^2 f(\vec{x}) \succeq \alpha I$ for all $\vec{x} \in \R^{n}$.
\end{definition}
\begin{remark}
  We say that $A \preceq B$ if $B - A \succeq 0$, i.e. $B - A$ is a positive semi-definite matrix.
  Equivalently, $\nabla^2 f(\vec{x}) \preceq \beta I$ means that all of the eigenvalues of $\nabla^2 f(\vec{x})$ are $\leq \beta$.
\end{remark}
So if a function is both $\alpha$-strongly convex and $\beta$-smooth then:
\[
  \alpha I \preceq \nabla^2 f(\vec{x}) \preceq \beta I
\]
\begin{theorem}
  If $f$ is $\beta$-smooth, then:
  \[
    f(\vec{y}) \leq f(\vec{x}) + \nabla f(\vec{x})^{\trans}(\vec{y} - \vec{x}) + \frac{\beta}{2}\norm{\vec{y} - \vec{x}}^2
  \]
  and if $f$ is $\alpha$-strongly convex, then:
  \[
    f(\vec{y}) \geq f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x}) + \frac{\alpha}{2}\norm{\vec{y} - \vec{x}}^2
  \]
\end{theorem}
\begin{remark}[Intuitition]
  In 1D, this is saying that we can trap the function below and above by a quadratic function.
\end{remark}
\begin{proof}
  Using the multivariate Taylor series:
  \[
    f(\vec{y}) = f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x}) + \frac{1}{2}(\vec{y} - \vec{x})^{\trans} \nabla^2 f(\vec{z}) (\vec{y} - \vec{x})
  \]
  where $\vec{z}$ is some point on the line segment joining $\vec{x}$ and $\vec{y}$.

  Now observe that:
  \begin{align*}
    \frac{1}{2}(\vec{y} - \vec{x})^{\trans} (\alpha I) (\vec{y} - \vec{x}) &\leq \frac{1}{2}(\vec{y} - \vec{x})^{\trans} \nabla^2 f(\vec{z}) (\vec{y} - \vec{x}) \leq \frac{1}{2}(\vec{y} - \vec{x})^{\trans} (\beta I) (\vec{y} - \vec{x}) \\
    \frac{\alpha}{2} \norm{\vec{y} - \vec{x}}^2 &\leq \frac{1}{2}(\vec{y} - \vec{x})^{\trans} \nabla^2f(\vec{z})(\vec{y} - \vec{x}) \leq \frac{\beta}{2} \norm{\vec{y} - \vec{x}}^2
  \end{align*}
  and so substituting back into the taylor series yields the result.
\end{proof}
\begin{corollary}
  For a $\beta$-smooth $f$ we have: \label{bSmoothCoro}
  \[
    f\left(\vec{x} - \frac{1}{\beta} \nabla f(\vec{x})\right) \leq f(\vec{x}) - \frac{1}{2\beta}\norm{\nabla f(\vec{x})}^2
  \]
\end{corollary}
\begin{remark}[Intuition]
  This gives us a step size to use for gradient descent as it guarantees that our step size does not ``overshoot''.
\end{remark}
\begin{proof}
  Follows by substituting $\vec{y} = \vec{x} - \frac{1}{\beta}\nabla f(\vec{x})$.
\end{proof}
\begin{corollary}
  For an $\alpha$-strongly convex function $f$, let $f(\vec{x}^{*})$ be the minimum of $f$.
  Then for any $\vec{x}$, we have: \label{aStrongCoro}
  \[
    f(\vec{x}^{*}) \geq f(\vec{x}) - \frac{1}{2\alpha} \norm{\nabla f(\vec{x})}^2
  \]
  or equivalently:
  \[
    f(\vec{x}) - f(\vec{x}^{*}) \leq \frac{1}{2\alpha} \norm{\nabla f(\vec{x})}^2
  \]
\end{corollary}
\begin{proof}
  We have:
  \[
    f(\vec{y}) \geq f(\vec{x}) + \nabla f(\vec{x})^{\trans}(\vec{y}  - \vec{x}) + \frac{\alpha}{2} \norm{\vec{y} - \vec{x}}^2
  \]
  Taking minima of both sides:
  \[
    \min_{\vec{y} \in \R^{n}} f(\vec{y}) \geq \min_{\vec{y} \in \R^{n}} \left[f(\vec{x}) + \nabla f(\vec{x})^{\trans}(\vec{y}  - \vec{x}) + \frac{\alpha}{2} \norm{\vec{y} - \vec{x}}^2\right]
  \]
  The left hand side is just $\min_{\vec{y} \in \R^{n}} f(\vec{y}) = f(\vec{x}^{*})$.
  For the right hand side:
  \[
    \nabla_{\vec{y}} \left(f(\vec{x}) + \nabla f(\vec{x})^{\trans}(\vec{y} - \vec{x}) + \frac{\alpha}{2}\norm{\vec{y} - \vec{x}}^2\right) = \nabla f(\vec{x}) + \alpha(\vec{y} - \vec{x})
  \]
  To minimise, we set this to $\vec{0}$ to give:
  \[
    \vec{y}^{*} = \vec{x} - \frac{1}{\alpha} \nabla f(\vec{x})
  \]
  Substituting back in to find the value at the minimum:
  \[
    \min_{\vec{y} \in \R^{n}} \left[f(\vec{x}) + \nabla f(\vec{x})^{\trans}(\vec{y}  - \vec{x}) + \frac{\alpha}{2} \norm{\vec{y} - \vec{x}}^2\right] = f(\vec{x}) - \frac{1}{2\alpha}\norm{\nabla f(\vec{x})}^2
  \]
  and so:
  \[
    f(\vec{x}^{*}) \geq f(\vec{x}) - \frac{1}{2\alpha}\norm{\nabla f(\vec{x})}^2
  \]
\end{proof}
\begin{remark}
  So if $\norm{\nabla f(\vec{x})}^2 \leq 2\alpha \varepsilon$, then $\vec{x}^{*}$ is optimal up to an $\varepsilon$ error, i.e.
  \[
    f(\vec{x}) - f(\vec{x}^{*}) \leq \varepsilon
  \]
\end{remark}
So we now know what step size we should take and what our stopping threshold should be.
\subsection{Convergence of Gradient Descent}
\begin{theorem}
  Let $f$ be $\alpha$-strongly convex and $\beta$-smooth and suppose $f$ is minimised at $\vec{x}^{*}$.
  Then gradient descent with $\eta_t = \frac{1}{\beta}$ satisfies:
  \begin{align*}
    f(\vec{x}_T) - f(\vec{x}^{*}) &\leq \left(1 - \frac{\alpha}{\beta}\right)^T (f(\vec{x}_0) - f(\vec{x}^{*})) \\
                                  &\leq e^{-\frac{\alpha T}{\beta}} (f(\vec{x}_0) - f(\vec{x}^{*}))
  \end{align*}
\end{theorem}
\begin{remark}
  The second inequality follows from that $e^{-x} \geq 1- x$.

  We like this inequality as we can then take logarithms to find $T$ that will give an $\varepsilon$-optimal point:
  \[
    T \geq \frac{\beta}{\alpha} \log \left(\frac{f(\vec{x}_0) - f(\vec{x}^{*})}{\varepsilon}\right)
  \]
\end{remark}
\begin{proof}
  \begin{align*}
    f(\vec{x}_{t + 1}) - f(\vec{x}^{*}) &\leq f(\vec{x}_t) - f(\vec{x}^{*}) - \frac{1}{2\beta} \norm{\nabla f(\vec{x}_t)}^2 \text{ (by \cref{bSmoothCoro})}\\
                                        &\leq f(\vec{x}_t) - f(\vec{x}^{*}) - \frac{\alpha}{\beta} (f(\vec{x}_t) - f(\vec{x}^{*})) \text{ (by \cref{aStrongCoro})}\\
                                        &= \left(1 - \frac{\alpha}{\beta}\right) (f(\vec{x}_t) - f(\vec{x}^{*})) \\
                                        &= \cdots \\
                                        &= \left(1 - \frac{\alpha}{\beta}\right)^{T} (f(\vec{x}_0) - f(\vec{x}^{*}))
  \end{align*}
\end{proof}
\begin{example}
  Consider $f: \R^{2} \to \R$ given by:
  \[
    f(\vec{x}) = \frac{1}{2}(x^2_1 + 100x^2_2)
  \]
  We have:
  \[
    \nabla^2 f(\vec{x}) = \begin{pmatrix}
    1 & 0 \\
    0 & 100 \\
    \end{pmatrix}\ \forall \vec{x} \in \R^2
  \]
  We see that:
  \[
    \begin{pmatrix}
    1 & 0 \\
    0 & 1 \\
    \end{pmatrix} \preceq \begin{pmatrix}
    1 & 0 \\
    0 & 100 \\
    \end{pmatrix} \preceq \begin{pmatrix}
    100 & 0 \\
    0 & 100 \\
    \end{pmatrix}
  \]
  So we can take $\alpha = 1$ and $\beta = 100$ so:
  \[
    1 - \frac{\alpha}{\beta} = 1 - \frac{1}{100} = 0.99
  \]
  $\beta = 100$ is quite large so we might want to rescale our axes so that it is not biased towards some direction.
  That is, currently, the algorithm will take small steps in general so that it doesn't overshoot when moving in the $x_2$ direction, however this means that it is taking steps that are too small in the $x_1$ direction.
\end{example}
\section{Second Order Methods}
Consider the second order Taylor approximation at $\vec{x}$:
\[
  f(\vec{y}) = f(\vec{x}) + \nabla f(\vec{x})^{\trans} (\vec{y} - \vec{x}) + \frac{1}{2} (\vec{y} - \vec{x})^{\trans} \nabla^2 f(\vec{x}) (\vec{y} - \vec{x})
\]
Suppose we change the update rule to the following:
\[
  \vec{x}_{t + 1} = \operatorname{argmin}_y [f(\vec{x}_t) + \nabla f(\vec{x}_t)^{\trans} (\vec{y} - \vec{x}_t) + \frac{1}{2} (\vec{y} - \vec{x}_t)^{\trans} \nabla^2 f(\vec{x}_t) (\vec{y} - \vec{x}_t)]
\]
To find the argmin of this, since it is quadratic, we need only consider when grad with respect to $\vec{y}$ of the function is $\vec{0}$:
\[
  \nabla_{\vec{y}} (f(\vec{x}_t) + \nabla f(\vec{x}_t)^{\trans} (\vec{y} - \vec{x}_t) + \frac{1}{2} (\vec{y} - \vec{x}_t)^{\trans} \nabla^2 f(\vec{x}_t) (\vec{y} - \vec{x}_t)) = \nabla f(\vec{x}_t) + \nabla^2f(\vec{x}_t) (\vec{y} - \vec{x}_t)
\]
Setting this to $\vec{0}$ gives:
\[
  \vec{y} = \vec{x}_t - (\nabla^2 f(\vec{x}_t))^{-1} \nabla f(\vec{x}_t)
\]
\subsection{Newton's Method}
Newton's method uses the update rule from the above, that is:
\[
  \vec{x}_{t + 1} = \vec{x}_t - (\nabla^2 f(\vec{x}_t))^{-1} \nabla f(\vec{x}_t)
\]
In one-dimension, this is a root finding method for $f'(x)$.
Suppose $f'(x) = g(x)$:
\[
  x_{t + 1} = x_t - \frac{g(x_t)}{g'(x_t)}
\]
Iterating this allows us to numerically find a root for $f'(x)$.
\begin{definition}[Matrix Norm]
  For $A \in \R^{n\times n}$, its \textit{norm} denoted $\norm{A}$ is the smallest $a > 0$ such that:
  \[
    \norm{A\vec{z}}_{2} \leq a \norm{\vec{z}}_{2}\ \forall \vec{z} \in \R^{n}
  \]
  where $\norm{\cdot}_2$ is the Euclidean norm.
\end{definition}
\begin{remark}
  For positive semi-definite $A$, its norm is the largest eigenvalue.
  Intuitively, the matrix norm upper bounds how much the matrix can ``stretch'' a vector in any direction.
\end{remark}
\begin{definition}[$\ell$-Lipschtiz Hessian]
  The Hessian $\nabla^2 f(\vec{x})$ of a function $f(\vec{x})$ is \textit{$\ell$-Lipschitz} if:
  \[
    \norm{\nabla^2 f(\vec{x}) - \nabla^2 f(\vec{y})} \leq \ell \norm{\vec{x} - \vec{y}}_{2}
  \]
\end{definition}
\begin{theorem}
  if $f$ is a twice-differentiable $\alpha$-strongly convex function with $\ell$-Lipschitz Hessian, then Newton's method applied to $f$ satisfies:
  \[
    f(\vec{x}_k) - f(\vec{x}^{*}) \leq \frac{2\alpha^3}{\ell^2}\left(\frac{\ell}{2\alpha^2}\norm{\nabla f(\vec{x}_0)}_2\right)^{2^{k + 1}}
  \]
\end{theorem}
\begin{proof}
  Provided without proof.
\end{proof}
\begin{remark}
  This only informs us about convergence if the bracketed term is less than 1, i.e. $\norm{\nabla f(\vec{x}_0)}_2 < \frac{2\alpha^2}{\ell}$.
  That is, we need to start at a point $\vec{x}_0$ that is sufficiently good to guarantee convergence.
  To achieve a sufficiently good starting point, we can sometimes use gradient descent and then switch to Newton's method.
\end{remark}
\subsection{Barrier Method}
Suppose we want to solve:
\begin{mini*}
  {\vec{x} \in \R^{n}}{f(\vec{x})}{}{}
  \addConstraint{\vec{a}_i^{\trans} \vec{x}}{\leq b_i \text{ for } i \in \{1, \ldots, m\}}
\end{mini*}
Here, instead of optimising on all of $\R^{n}$, these constraints bound us to a particular subset of $\R^{n}$.
For example in $\R^2$, each of these constraints is a straight line.

We can however modify the problem so that it we are in fact optimising over all $\R^{n}$:
\begin{mini*}
  {\vec{x} \in \R^{n}}{f(\vec{x}) + \sum_{i = 1}^{m} \phi(\vec{a}_i^{\trans} \vec{x} - b_i)}{}{}
\end{mini*}
At first, we might try:
\[
  \phi(t) = \begin{cases}
  t & \text{ if } t \leq 0 \\
  + \infty & \text{ if } t > 0
  \end{cases}
\]
but this would not stop Newton's method from potentially leaving the region.
A common choice is instead the \textit{logarithmic barrier function} $\phi(x) = -\log(-x)$ as this shoots off to $\infty$ as $x \to 0^{-}$ but does so continuously and not as a jump.

We also introduce a parameter $t > 0$ that allows us to shift the focus between our new problem, and the original problem of minimising $f(\vec{x})$, but now with the ``barrier'' around the feasible region:
\begin{mini*}
  {\vec{x} \in \R^{n}}{tf(\vec{x}) - \sum_{i = 1}^{m} \log(-(\vec{a}_i^{\trans} \vec{x} - b_i))}{}{}
\end{mini*}
As $t \to \infty$, the problem shifts back to minimising $f(\vec{x})$.
This is the principle of the barrier method.
\subsubsection{Steps for the Barrier Method}
\begin{enumerate}
  \item Find a strictly feasible $\vec{x}$ and set $t > 0$ and $\alpha > 1$.
  \item Repeat the following:
    \begin{enumerate}
      \item Compute $\vec{x}^{*}(t)$ by minimising:
        \[
         tf(\vec{x}) - \sum_{i = 1}^{m} \log(-(\vec{a}_i^{\trans} \vec{x} - b_i))
        \]
        using Newton's method with $\vec{x}$ as the starting point.
      \item Update $\vec{x} \mapsto \vec{x}^{*}(t)$ and $t \mapsto \alpha t$, i.e. take the new starting point to be the optimal solution for the current $t$ and then increase $t$ slightly.
      \item Stop when $t$ is large enough.
    \end{enumerate}
\end{enumerate}
