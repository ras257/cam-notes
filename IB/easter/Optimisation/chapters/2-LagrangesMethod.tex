\chapter{Lagrange's Method}
\section{Setup}
\begin{definition}[Lagrangian]
  We define the \textit{Lagrangian} as:
  \[
    L(\vec{x}, \vec{\lambda}) = f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b}) \]
  for some vector $\vec{\lambda} \in \R^{n}$
\end{definition}
Suppose that we start with the following problem for $f: \R^{n} \to \R^{n}$:
\begin{mini*}
  {\vec{x} \in \mathcal{X}}{f(\vec{x})}{}{}
  \addConstraint{h(\vec{x})}{= \vec{b}}
\end{mini*}
where $\vec{\lambda}, \vec{b} \in \R^{m}$ and $h: \R^{n} \to \R^{m}$.
We can instead try solving:
\begin{mini*}
  {\vec{x} \in \mathcal{X}}{L(\vec{x}, \vec{\lambda})}{}{}
\end{mini*}
\begin{theorem}[Lagrange Sufficiency Theorem]
  Let $\vec{x}^{*} \in \mathcal{X}$ and $h(\vec{x}^{*}) = \vec{b}$.
  If we can find $\vec{\lambda}^{*} \in \R^{m}$ such that:
  \[
    L(\vec{x}^{*}, \vec{\lambda}^{*}) = \min_{\vec{x} \in \mathcal{X}} L(\vec{x}, \vec{\lambda}^{*})
  \]
  Then $\vec{x}^{*}$ is an optimal solution of the original problem.
\end{theorem}
\begin{proof}
  Suppose we are able to find $\vec{\lambda}^{*}$ as above.

  Since $\vec{x}^{*} \in \mathcal{X}(\vec{b})$ we have that:
  \[
    \min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x}) \leq f(\vec{x}^{*})
  \]

  To prove the reverse inequality, since $h(\vec{x}) - \vec{b} = \vec{0}\ \forall \vec{x} \in \mathcal{X}(\vec{b})$, we can rewrite:
  \begin{align*}
    \min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x}) &= \min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x}) - \vec{\lambda}^{\trans} (h(\vec{x}) - \vec{b}) \\
                                                       &\geq \min_{\vec{x} \in \mathcal{X}} \underbrace{f(\vec{x}) - {\vec{\lambda}^{*}}^{\trans} (h(\vec{x}) - \vec{b})}_{=L(\vec{x}, \vec{\lambda}^{*})} \\
                                                       &= f(\vec{x}^{*}) - {\vec{\lambda}^{*}}^{\trans}(\underbrace{h(\vec{x}^{*}) - \vec{b}}_{= \vec{0}}) \\
                                                       &= f(\vec{x}^{*})
  \end{align*}
\end{proof}
\begin{example}
  Consider the problem:
  \begin{mini*}
    {\vec{x}}{-x_1 - x_2 + x_3}{}{}
    \addConstraint{x^2_1 + x^2_2}{= 4}
    \addConstraint{x_1 + x_2 + x_3}{=1}
  \end{mini*}
  Construct the Lagrangian:
  \begin{align*}
    L(\vec{x}, \vec{\lambda}) &= L(x_1, x_2, x_3, \lambda_1, \lambda_2) \\
                              &= - x_1 - x_2 + x_3 - \lambda_1 (x^2_1 + x^2_2 - 4) - \lambda_2 (x_1 + x_2 + x_3 - 1) \\
                              &= [(-1-\lambda_2)x_1 -\lambda_1 x^2_1] + [(-1-\lambda_2)x_2 -\lambda_1 x^2_2] + [(1 - \lambda_2)x_3] + 4\lambda_1 + \lambda_2
  \end{align*}
  This is just a sum of quadratic and linear functions of $x_1, x_2, x_3$ and so is quite easy to minimise.
  We try to solve:
  \begin{mini*}
    {\vec{x}}{L(\vec{x}, \vec{\lambda})}{}{}
    \addConstraint{\text{No constraints}}{}
  \end{mini*}
  For this problem to make sense (i.e. to be able to find $\vec{x}$ which solve this), we must have $\lambda_2 = 1$ otherwise the function would be unbounded below as we take $x_3 \to -\infty$.
  Then Lagrangian is then:
  \[
    L(\vec{x}, \vec{\lambda}) = (-2x_1 -\lambda_1 x^2_1) + (-2 x_2 - \lambda_1 x^2_2) + 4\lambda_1 + 1
  \]
  We also must have $\lambda_1 \leq 0$ otherwise again the function is unbounded below.
  On minimising these quadratics, we obtain that:
  \[
    x_1 = - \frac{1}{\lambda_1} \text{ and } x_2 = - \frac{1}{\lambda_1}
  \]
  So we know that if we take:
  \[
    (\lambda_1, 1) \mapsto \left(-\frac{1}{\lambda_1}, - \frac{1}{\lambda_1}, x_3\right)
  \]
  for $\lambda_1 \leq 0$.
  We now need to tweak $\lambda_1$ so that the point
  \[
    \left(-\frac{1}{\lambda_1}, - \frac{1}{\lambda_1}, x_3\right)
  \]
  becomes feasible.
  That is, we need:
  \[
    \frac{1}{\lambda^2_1} + \frac{1}{\lambda^2_1} = 4 \text{ and } -\frac{1}{\lambda_1} - \frac{1}{\lambda_1} + x_3 = 1
  \]
  which is solved by $\lambda_1 = - \frac{1}{\sqrt{2}}$.
  This gives:
  \[
    \vec{x}^{*} = (\sqrt{2}, \sqrt{2}, 1 - 2\sqrt{2}) \text{ and } \vec{\lambda}^{*} = \left(-\frac{1}{\sqrt{2}}, 1\right)
  \]
  So using Lagrange sufficiency we conclude that $\vec{x}^{*}$ solves the original problem.
\end{example}
\section{Algorithm For The Lagrange Method}
Consider the problem:
\begin{mini*}
  {\vec{x} \in \mathcal{X}}{f(\vec{x})}{}{}
  \addConstraint{h(\vec{x}) \leq \vec{b}}{}
\end{mini*}
\subsubsection{The Algorithm}
\begin{enumerate}
  \item Add slack variables to consider the new problem:
    \begin{mini*}
      {\vec{x} \in \mathcal{X},\ \vec{s} \geq \vec{0}}{f(\vec{x})}{}{}
      \addConstraint{h(\vec{x}) + \vec{s} = \vec{b}}{}
    \end{mini*}
  \item Set up:
    \[
      L(\vec{x}, \vec{s}, \vec{\lambda}) = f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) + \vec{s} - \vec{b})
    \]
  \item We also define the set $\Lambda$ as:
    \[
      \Lambda = \left\{\vec{\lambda} \in \R^{m}: \min_{\vec{x} \in \mathcal{X}(\vec{s})} L(\vec{x}, \vec{s}, \vec{\lambda}) > -\infty\right\}
    \]
  \item For each $\vec{\lambda} \in \Lambda$, we can find $\vec{x}^{*}(\vec{\lambda})$ and $s^{*}(\vec{\lambda})$ such that:
    \[
      L(\vec{x}^{*}(\vec{\lambda}),  \vec{s}^{*}(\vec{\lambda})) = \min_{\substack{\vec{x} \in \mathcal{X} \\ \vec{s} \geq 0}} L(\vec{x}, \vec{s}, \vec{\lambda})
    \]
  \item If we can find $\vec{\lambda}^{*}$ such that $(\vec{x}^{*}(\vec{\lambda}^{*}), \vec{s}^{*}(\vec{\lambda}^{*}))$ is feasible, that is $\vec{x}^{*}(\vec{\lambda}^{*}) \in \mathcal{X}$ satisfies $h(\vec{x}^{*}) + \vec{s}^{*} = \vec{b}$ and $\vec{s}^{*} \geq 0$, then $\vec{x}^{*}(\vec{\lambda}^{*})$ is optimal.
\end{enumerate}
\subsubsection{Complimentary Slackness}
Recall the Lagrangian from step \textbf{(ii)}:
\[
  L(\vec{x}, \vec{s}, \vec{\lambda}) = f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b}) - \vec{\lambda}^{\trans} \vec{s}
\]
If a particular $\lambda_i < 0$, then we observe that the Lagrangian is minimised at $s_i = 0$.
In effect, for each $i$, we must have $\lambda_i s_i = 0$ in the optimal solution.
\begin{example}
  Consider the problem:
  \begin{mini*}
    {\vec{x}}{x_1 - 3x_2}{}{}
    \addConstraint{x^2_1 + x^2_2}{\leq 4}
    \addConstraint{x_1 + x_2}{\leq 2}
  \end{mini*}
  We setup the Lagrangian as:
  \begin{align*}
    L(\vec{x}, \vec{s}, \vec{\lambda}) &= x_1 - 3x_2 - \lambda_1(x^2_1 + x^2_2 + s_1 - 4) - \lambda_2(x_1 + x_2 + s_2 - 2) \\
                                       &= [(1 - \lambda_2)x_1 - \lambda_1 x^2_1] + [(-3 - \lambda_2)x_2 - \lambda_1 x^2_2] - \lambda_1 s_1 - \lambda_2 s_2 + 4\lambda_1 + 2\lambda_2
  \end{align*}
  \begin{proofcases}
    \begin{case}{$\lambda_1 < 0,\ \lambda_2 < 0$}
      Complimentary slackness then tells us that $s_1 = s_2 = 0$ so after optimising using the derivative we obtain the system:
      \begin{align*}
        1 - 2\lambda_1 x_1 - \lambda_2 &= 0 \\
        -3 -2 \lambda_1 x_2 - \lambda_2 &= 0 \\
        x^2_1 + x^2_2 &= 4 \\
        x_1 + x_2 &= 2 \\
      \end{align*}
      Firstly, from the last two equations we see that either $(x_1, x_2) = (0, 2)$ or $(x_1, x_2) = (2, 0)$.
      In the case of $(x_1, x_2) = (0, 2)$, we must have $(\lambda_1, \lambda_2) = (1, -3)$ and in the case of $(x_1, x_2) = (2, 0)$, we must have $(\lambda_1, \lambda_2) = (-1, 1)$.
      However this is inconsistent with $\lambda_1 < 0$ and $\lambda_2 < 0$ so this case cannot be consistent.
    \end{case}
  \end{proofcases}
  We have $2^{2}$ cases we can check as for each of $\lambda_i$ we pick either $\lambda_i = 0$ or $\lambda_i > 0$.
  As soon as we find a case which works, we can stop.
  If you never find a case which works, then the Lagrange method simply does not work, even if the problem is well formed.
\end{example}
\section{Duality}
Consider the \textit{primal} problem:
\begin{mini*}
  {\vec{x} \in \mathcal{X}}{f(\vec{x})}{}{}
  \addConstraint{h(\vec{x})}{= \vec{b}}
\end{mini*}
We can construct a \textit{dual} problem from this.
\begin{definition}[Dual Objective Function]
  The \textit{dual objective function}, $g: \Lambda \to \R$ is defined as:
  \[
    g(\vec{\lambda}) = \min_{\vec{x} \in \mathcal{X}} L(\vec{x}, \vec{\lambda})
  \]
  for $\Lambda = \{\vec{\lambda}: \min_{\vec{x} \in \mathcal{X}} L(\vec{x}, \vec{\lambda}) > -\infty\}$.
\end{definition}
\begin{definition}[Dual Problem]
  The \textit{dual problem} is:
  \begin{maxi*}
    {\vec{\lambda} \in \Lambda}{g(\vec{\lambda})}{}{}
  \end{maxi*}
\end{definition}
For every fixed $\vec{x}$, the Lagrangian is linear in $\vec{\lambda}$:
\[
  L(\vec{x}, \vec{\lambda}) = f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b})
\]
We can show that the minimum of linear functions is concave and so $g(\vec{\lambda})$ is concave.
Thus we can maximise $g(\vec{\lambda})$ using gradient descent, Newton's method, or any other method for maximising concave functions.
\begin{theorem}[Weak Duality]
  If $\vec{x} \in \mathcal{X}(\vec{b})$ and $\vec{\lambda} \in \Lambda$, then $f(\vec{x}) \geq g(\vec{\lambda})$ and so in particular:
  \[
    \min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x}) \geq \max_{\vec{\lambda} \in \Lambda} g(\vec{\lambda})
  \]
  Importantly, solving the dual problem gives us a \textbf{lower bound} for the minimum of $f(\vec{x})$.
\end{theorem}
\begin{proof}
  For any $\vec{\lambda} \in \Lambda$ and $\vec{x} \in \mathcal{X}(\vec{b})$:
  \begin{align*}
    f(\vec{x}) &\geq \min_{\vec{x} \in \mathcal{X}(\vec{b})} (\vec{x}) \\
               &= \min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b}) \\
               &\geq \min_{\vec{x} \in \mathcal{X}} f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b}) \\
               &= \min_{\vec{x} \in \mathcal{X}} L(\vec{x}, \vec{\lambda}) \\
               &= g(\vec{\lambda})
  \end{align*}
\end{proof}
\begin{definition}[Strong Duality]
  The gap between $\min_{\vec{x} \in \mathcal{X}(\vec{b})} f(\vec{x})$ and $\max_{\vec{\lambda} \in \Lambda} g(\vec{\lambda})$ is called the \textit{duality gap}.
  If the duality gap is 0, then we say that \textit{strong duality} holds.
\end{definition}
\begin{theorem}
  The Lagrange method works if and only if strong duality holds.
\end{theorem}
\begin{proof}
  \begin{proofdirection}{Suppose the Lagrange method works}
    This means that we have $(\vec{x}^{*}, \vec{\lambda}^{*})$ such that $\argmin_{\vec{\lambda}} (\vec{x}, \vec{\lambda}^{*}) = \vec{x}^{*}$ and $h(\vec{x}^{*}) = \vec{b}$.
    \begin{align*}
      g(\vec{\lambda}^{*}) &= \min_{\vec{x} \in \mathcal{X}}  f(\vec{x}) - {\vec{\lambda}^{*}}^{\trans} (h(\vec{x}) - \vec{b}) \\
                           &= f(\vec{x}^{*}) - {\vec{\lambda}^{*}}^{\trans}(\underbrace{h(\vec{x}^{*}) - \vec{b}}_{\vec{0}}) \\
                           &= f(\vec{x}^{*})
    \end{align*}
    and so strong duality holds.
  \end{proofdirection}
  \begin{proofdirection}{Suppose strong duality holds}
    So $\exists \vec{x} \in \mathcal{X}(\vec{b})$, $\vec{\lambda} \in \Lambda$ such that $f(\vec{x}) = g(\vec{\lambda})$.
    Then the Lagrange method works using this $\vec{\lambda}$.
  \end{proofdirection}
\end{proof}
\begin{definition}[Supporting Hyperplane]
  A function $\phi: \R^{m} \to \R$ is said to have a \textit{supporting hyperplane} at a point $\vec{b} \in \R^{m}$ if there exists $\vec{\lambda}$ such that for all $\vec{c} \in \R^{m}$:
  \[
    \phi(\vec{x}) \geq \phi(\vec{b}) + \vec{\lambda}^{\trans}(\vec{c} - \vec{b})
  \]
\end{definition}
\begin{remark}
  \begin{itemize}
    \item Saying there is a supporting hyperplane is the same as saying that there exists a tangent at that point which lies below the curve.
  \end{itemize}
\end{remark}
\begin{definition}[Value Function]
  Define the \textit{value function} $\phi: \R^{m} \to \R$ as:
  \[
    \phi(\vec{c}) = \min_{\vec{x} \in \mathcal{X}(\vec{c})} f(\vec{x})
  \]
  where $\mathcal{X}(\vec{c}) = \{\vec{x} \in \mathcal{X}: h(\vec{x}) = \vec{c}\}$.
\end{definition}
\begin{remark}
  This is a function which outputs the minimum of the problem:
  \begin{mini}
    {\vec{x} \in \mathcal{X}}{f(\vec{x})}{\label{hyperProb}}{}
    \addConstraint{h(\vec{x})}{= \vec{c}}
  \end{mini}
  for a given $\vec{c} \in \R^{m}$ instead of just a fixed $\vec{b}$.
\end{remark}
\begin{theorem}
  \Cref{hyperProb} satisfies strong duality if the value function $\phi$ has a supporting hyperplane at $\vec{b}$.

  Conversely, if strong duality holds for a suitable $\vec{\lambda}$, i.e $g(\vec{\lambda}) = \phi(\vec{b})$, then $\vec{\lambda}$ is the normal for the supporting hyperplane to $\phi$ at $\vec{b}$.
\end{theorem}
\begin{proof}
  \begin{proofdirection}{Suppose $\phi$ has supporting hyperplane at $\vec{b}$}
    Let the hyperplane have normal $\vec{\lambda}$ at $\vec{b}$.
    This means that:
    \[
      \phi(\vec{x}) \geq \phi(\vec{b}) + \vec{\lambda}^{\trans}(\vec{c} - \vec{b})
    \]
    We then have:
    \begin{align*}
      g(\vec{\lambda}) &= \min_{\vec{x} \in \mathcal{X}} f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{b}) \\
                       &= \min_{\vec{c}} \min_{\vec{x} \in \mathcal{X}(\vec{c})} f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{c}) - \vec{\lambda}^{\trans}(\vec{c} - \vec{b})
    \end{align*}
    since $\mathcal{X} = \bigcup_{\vec{c}} \mathcal{X}(\vec{c})$, i.e. we partition the minimum over $\mathcal{X}$ into sections $\mathcal{X}(\vec{c})$.

    We then have:
    \[
      \min_{\vec{x} \in \mathcal{X}(\vec{c})} f(\vec{x}) = \phi(\vec{c})
    \]
    and $h(\vec{x}) - \vec{c} = \vec{0}$ for $\vec{x} \in \mathcal{X}(\vec{c})$ so:
    \begin{align*}
      g(\vec{\lambda}) &= \min_{\vec{c}} \phi(\vec{c}) - \vec{\lambda}^{\trans}(\vec{c} - \vec{b}) \\
                       &\geq \phi(\vec{b})
    \end{align*}
    By weak duality $g(\vec{\lambda}) \leq \phi(\vec{b})$, thus $g(\vec{\lambda}) = \phi(\vec{b})$, so we have strong duality.
  \end{proofdirection}
  \begin{proofdirection}{Suppose strong duality holds}
    So we have $g(\vec{\lambda}) = \phi(\vec{b})$ for some $\vec{\lambda}$.
    Then:
    \begin{align*}
      \phi(\vec{b}) = g(\vec{\lambda}) &= \min_{\vec{x} \in \mathcal{X}}  f(\vec{x}) - \vec{\lambda}^{\trans} (h(\vec{x}) - \vec{b}) \\
                                       &= \min_{\vec{x} \in \mathcal{X}}  f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{c}) - \vec{\lambda}^{\trans}(\vec{c} - \vec{b}) \\
                                       &\leq \phi(\vec{c}) - \vec{\lambda}^{\trans}(\vec{c} - \vec{b})
    \end{align*}
    because:
    \begin{align*}
      \min_{\vec{x} \in \mathcal{X}} f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{c}) &\leq \min_{\vec{x} \in \mathcal{X}(\vec{c})}  f(\vec{x}) - \vec{\lambda}^{\trans}(h(\vec{x}) - \vec{c}) \\
                                                                                               &= \phi(\vec{c})
    \end{align*}
  \end{proofdirection}
\end{proof}
So we have that:
\[
  \text{Lagrange works} \iff \text{Strong duality} \iff \phi \text{ has supporting hyperplane at $\vec{b}$}
\]
Thus if we can show that $\phi$ is convex, it will have a supporting hyperplane at any point, in particular at $\vec{b}$, and so the Lagrange method will work.
\begin{theorem}
  If $\mathcal{X}$ is a convex set, $f: \R^{n} \to \R$ is convex, and $h_j: \R^{n} \to \R$ are convex functions for $j \in \{1, \ldots, m\}$, then the problem:
  \begin{mini*}
    {\vec{x} \in \mathcal{X}}{f(\vec{x})}{}{}
    \addConstraint{h_j(\vec{x})}{\leq b_j,\ j \in \{1, \ldots, m\}}
  \end{mini*}
  has a convex value function $\phi$.
\end{theorem}
