\chapter{Principle of Least Action}
Consider a system of $N$ particles in 3D with positions $\vec{x}_1, \ldots, \vec{x}_n$.
We can consider the positions of the particles at a single point as a $3N$ dimensional point in \textit{configuration space}:
\[
  (\vec{x}_1, \ldots, \vec{x}_N)
\]
It is sometimes also useful to use non-Cartesian coordinates of the space:
\[
  \vec{q} = (\vec{q}_1, \ldots, \vec{q}_n)
\]
There is a simple way to determine the equations of motion if we know the kinetic energy $T$ and potential energy $V$ in terms of $\vec{q}(t)$ and $\dot{\vec{q}}(t)$.
\begin{definition}[Lagrangian]
  We define the \textit{Lagrangian} as:
  \[
    L(\vec{q}, \dot{\vec{q}}, t) = T - V
  \]
\end{definition}
\begin{definition}[Action]
  The \textit{action} of a path $\vec{q}(t)$ in configuration space between $\vec{q}_A$ and time $t_A$, and $\vec{q}_B$ at time $t_B$ is:
  \[
    I[\vec{q}] = \int_{t_A}^{t_B} L(\vec{q}(t), \dot{\vec{q}}(t), t) \d{t}
  \]
\end{definition}
\begin{law}
  The principle of least action tells us that the actual path from $\vec{q}_A$ to $\vec{q}_B$ of the system extremities the action.
\end{law}
Since $\vec{q}_A$ and $\vec{q}_B$ are fixed, the boundary term in $\delta I$ vanishes and so we just recover the Euler-Lagrange equations.
Since $\vec{q}$ has $3N$ components, we will obtain $3N$ Euler-Lagrange equations:
\[
  \pderiv{L}{q_i} - \deriv{}{t} \pderiv{L}{\dot{q}_i} = 0 \text{ for } i \in \{1, \ldots, 3N\}
\]
\begin{example}
  If we take $N = 1$ and $\vec{q} = \vec{x}$ (i.e. we work in Cartesian coordinates), we know that $T = \frac{1}{2} m \dot{\vec{x}} \cdot \dot{\vec{x}}$ for fixed $m$ and we assume that we can write $V = V(\vec{x}, t)$. \label{n1LG}
  The Lagrangian is then:
  \[
    L = \frac{1}{2} m \dot{\vec{x}} \cdot \dot{\vec{x}} - V(\vec{x}, t)
  \]
  and so the Euler-Lagrange equations tells us that:
  \[
    \pderiv{V}{x_i} - \deriv{}{t}(m \dot{x}_i) = 0 \iff m \ddot{\vec{x}} = -\nabla V
  \]
  This is Newton's 2nd Law for a conservative force $\vec{F} = - \nabla V$.
\end{example}
\begin{proposition}
  If $L$ has no explicit $t$ dependence, then there exists a first integral:
  \[
    L - \sum_{i = 1}^{3N}  \dot{q}_i \pderiv{L}{\dot{q}_i} = - E
  \]
  for some constant $E$ for any solution of the Euler-Lagrange equations.
\end{proposition}
\begin{proof}
  \begin{align*}
    \deriv{L}{t} &= \deriv{}{t} L(\vec{q}(t), \dot{\vec{q}}(t), t) \\
                 &= \pderiv{L}{t} + \sum_{i = 1}^{3N} \left(\pderiv{L}{q_i} \dot{q}_i + \deriv{L}{\dot{q_i}} \ddot{q}_i\right) \\
                 &= \pderiv{L}{t} + \underbrace{\sum_{i = 1}^{3N} \left(\pderiv{L}{q_i} - \deriv{}{t} \pderiv{L}{\dot{q_i}}\right)}_{= 0} + \deriv{}{t}\left(\sum_{i = 1}^{3N} \dot{q}_i \pderiv{L}{\dot{q_i}}\right) \\
  \end{align*}
  Rearranging we have:
  \[
    \deriv{}{t}\left(L - \sum_{i = 1}^{3N} \dot{q}_i \pderiv{L}{\dot{q}_i}\right) = \pderiv{L}{t} = 0
  \]
  provided there is no explicit $t$ dependence.
\end{proof}
\begin{example}
  Continuing from \cref{n1LG}, we see that $V = V(x)$ and so we can find $E$ as:
  \begin{align*}
    -E &= \frac{1}{2} m \dot{\vec{x}} \cdot \dot{\vec{x}} - V - \sum_{i = 1}^{3} \dot{x}_i m \dot{x}_i \\
       &= \frac{1}{2} m \dot{\vec{x}} \cdot \dot{\vec{x}} - V(\vec{x}) \\
       &= - (T + V)
  \end{align*}
  So $E = T + V$ is the total energy.
\end{example}
