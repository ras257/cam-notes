\chapter{Power Series}
\section{Series and Sequences of Functions}
\begin{definition}[Sequence of Functions]
  We say $(f_n) = f_1, f_2, \ldots$ is a \textit{sequence of complex valued functions} on some set $X \subseteq \C$ if $f_n: X \to \C$ for each $n \in \N$.

  Further, if for some set $Y \subseteq X$ the numerical sequence $(f_n(x))_n$ converges for all $x \in Y$, then we can define a function:
  \[
    f: Y \to \C,\ x \mapsto \lim_{n \to \infty} f_n(x)
  \]
  and then we say $f_n \to f$ as $n \to \infty$ on the set $Y$.
\end{definition}
\begin{definition}[Series of Functions]
  Let $(f_n)$ be a sequence of functions as above.
  We then call $\sum_{n = 1}^{\infty} f_n$ a \textit{series of complex valued functions} on $X$.

  Further, if for each $x \in Y$, the numerical series $\sum_{n = 1}^{\infty} f_n(x)$ converges, then we can define a function:
  \[
    f: Y \to \C,\ x \mapsto \sum_{n = 1}^{\infty} f_n(x)
  \]
\end{definition}
If we know something about the continuity, differentiability, or integrability about the functions $f_i$, will those properties still hold for its sequence or series?
This is a subtle question, and essentially boils down to if we are able to swap limits, sums, and or integrals.
\begin{example}
  If $f_n$ are all continuous at $a$ and we want to show that $f(x) = \lim_{n \to \infty} f_n(x)$ (which is defined on some set $Y$ such that $a \in Y$) is continuous at $a$ also.

  We can write $f(a)$ and $\lim_{x \to a} f(x)$ as a pair of nested limits:
  \begin{align*}
    f(a) &= \lim_{n \to \infty} f_n(a) = \lim_{n \to \infty} \lim_{x \to a} f_n(x) \\
    \lim_{x \to a} f(x) &= \lim_{x \to a} \lim_{n \to \infty} f_n(x)
  \end{align*}
  So to conclude that $\lim_{x \to a} f(x) = f(a)$ (i.e. $f$ is continuous at $a$), we would need to show that:
  \[
    \lim_{x \to a} \lim_{n \to \infty} f_n(x) = \lim_{n \to \infty} \lim_{x \to a} f_n(x)
  \]
  Swapping these limits is non-trivial, for example, if we define $S_{m, n} = \frac{m}{n + m}$:
  \begin{align*}
    \lim_{n \to \infty} \lim_{m \to \infty} S_{m, n} = 1 \\
    \lim_{m \to \infty} \lim_{n \to \infty} S_{m, n} = 0
  \end{align*}
\end{example}
This will be explored properly in Analysis II.
For this course, we will just consider simpler series of functions called \textit{power series} where $f_n$ are of the form:
\[
  f_n(x) = c_n(x - a)^{n}
\]
\section{Basics of Power Series}
\subsection{Definitions}
\begin{definition}[Power Series]
  Let $a \in \C$ and let $(c_n)_{n \in \N_0}$ be a numerical sequence on $\C$.
  We call:
  \[
    \sum_{n = 0}^{\infty} c_n(x - a)^{n}
  \]
  a \textit{power series with centre $a$ and coefficients $c_n$}.
\end{definition}
\begin{example}[Taylor Series]
  If $f$ is a smooth function, then its Taylor series (\cref{taylorSeries}) about $a$ is a power series with coefficients:
  \[
    c_n = \frac{f^{(n)}(a)}{n!}
  \]
\end{example}
Tautologically, every power series must converge at its centre $a$.
The natural question is whether the set:
\[
  \left\{x \in \C: \sum_{n = 0}^{\infty} c_n(x - a)^{n} \text{ converges at $x$}\right\}
\]
contains any other points?
\subsection{Radius of Convergence}
\begin{lemma}
  If $\sum_{n=0}^{\infty} c_n (x - a)^{n}$ converges at a point $x \in \C$ and $|y - a| < |x - a|$ for some $y \in \C$, then $\sum_{n = 0}^{\infty} c_n(y - a)^{n}$ converges absolutely. \label{powerSeriesConvLemma}
\end{lemma}
\begin{remark}[Statement in words]
  If the power series converges at a point $x$, then it must converge absolutely at any points closer to the centre $a$ than $x$.
\end{remark}
\begin{proof}
  The proof idea here is to try to use a comparison to the geometric series.

  Fix $x \in \C$ such that $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ and fix $y \in \C$ such that $|y - a| < |x - a|$.
  Let $r = \frac{|y - a|}{|x - a|} < 1$, we then have:
  \begin{align*}
    |c_n(y - a)^{n}| &= |c_n(x - a)^{n}| \cdot \abs{\frac{y - a}{x - a}}^{n} \\
                     &\leq |c_n(x - a)^{n}| \cdot r^{n}
  \end{align*}
  By assumption, $\sum_{n = 1}^{\infty} c_n(x - a)^{n}$ converges, so by the $n$-th term test (\cref{nthTermTest}), $c_n(x - a)^{n} \to 0$ as $n \to \infty$.
  Thus $c_n(x - a)^{n}$ must be bounded so $\exists M > 0$ such that $|c_n(x - a)^{n}| \leq M\ \forall n$.
  Hence:
  \[
    |c_n(y - a)^{n}| \leq M \cdot r^{n}\ \forall n
  \]
  So using the comparison test (\cref{comparisonTest}) with a geometric series of ratio $r$, $\sum_{n = 0}^{\infty} |c_n(y - a)^{n}|$ converges and therefore $\sum_{n = 0}^{\infty} c_n(y - a)^{n}$ converges absolutely.
\end{proof}
\begin{proposition}[Radius of Convergence]
  If $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ is a power series, then $\exists R \in [0, \infty]$, such that:
  \begin{enumerate}
    \item $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ converges absolutely for $|x - a| < R$.
    \item If $R < \infty$, then $\sum_{n =0}^{\infty} c_n(x - a)^{n}$ diverges for $|x - a| > R$.
  \end{enumerate}
  We call $R$ the \textit{radius of convergence} (\textit{r.o.c}) of the power series.
\end{proposition}
\begin{remark}[Remarks]
  \begin{itemize}
    \item The above \textbf{does not} tell us what happens on the circle $|x - a| = R$.
      The power series could have wildly different behaviour at different points on this circle.
    \item If $R = 0$, then the series converges only at $a$.
    \item If $R = \infty$, then the series converges on all of $\C$.
  \end{itemize}
\end{remark}
\begin{proof}
  Let $A$ be the set of radii $r$ which have a point of convergence inside the closed disc of radius $r$ centered at $a$, that is:
  \[
    A = \left\{r \geq 0 : \exists x \in \C \text{ with } |x - a| = r \text{ s.t. } \sum_{n = 0}^{\infty} c_n(x - a)^{n} \text{ converges}\right\}
  \]
  We know that the power series always converges at $a$ so $0 \in A \implies A \neq \emptyset$.
  \begin{proofcases}
    \begin{case}{$A$ is unbounded}
      If $A$ is unbounded, then can set $R = \infty$ as given any point $y$, we can always find a point further from the centre where the power series converges so by \cref{powerSeriesConvLemma}, it must converge absolutely at $y$ and thus, since $y$ was arbitrary, it converges everywhere in $\C$.
    \end{case}
    \begin{case}{$A$ is bounded}
      Since $A$ is bounded, it must have a supremum so we set $R = \sup A$.
      By definition of the set $A$, $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ diverges for $|x - a| > R$.

      If $|y - a| < R = \sup A$, then, by definition of supremum, $\exists r \in A$ with $|y - a| < r \leq R$.
      Since $r \in A$, $\exists x \in \C$ with $|x - a| = r > |y - a|$ such that $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ converges.
      So by \cref{powerSeriesConvLemma}, $\sum_{i = 0}^{\infty} c_n(y - a)^{n}$ converges absolutely.
    \end{case}
  \end{proofcases}
  So in both cases, a suitable radius of convergence $R$ exists.
\end{proof}
To compute the radius of converges of a power series, we can adapt the usual tests for numerical series.
\begin{proposition}[Root Test for Power Series]
  If $(c_n)$ is a sequence of coefficients on $\C$ such that:
  \[
    L = \lim_{n \to \infty} \sqrt[n]{|c_n|}
  \]
  exists, then the power series with those coefficients has radius of convergence $R = \frac{1}{L}$, with the convention convention $L = 0 \to R = \infty$.
\end{proposition}
\begin{proof}
  Consider the power series $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$.
  For some fixed $x$, we have:
  \[
    \lim_{n \to \infty} \sqrt[n]{\abs{c_n(x - a)^{n}}} = |x - a|\lim_{n \to \infty} \sqrt[n]{\abs{c_n}} = |x - a|L
  \]
  So by the ratio test (\cref{ratioTest}), if $|x - a|L < 1$, then the power series converges absolutely and if $|x - a|L > 1$, the power series diverges by the $n$-th term test (\cref{nthTermTest}).

  In the case of $L = 0$, it must always converge so $R = \infty$, otherwise, it converges absolutely for $|x - a| < \frac{1}{L}$ and diverges for $|x - a| > \frac{1}{L}$ so its radius of convergence must be $R = \frac{1}{L}$.
\end{proof}
\begin{proposition}[Ratio Test for Power Series]
  If $(c_n)$ is a sequence of coefficients on $\C$ such that:
  \[
    L = \lim_{n \to \infty} \abs{\frac{c_{n + 1}}{c_n}}
  \]
  exists, then the power series with those coefficients has radius of convergence $R = \frac{1}{L}$, with convention the convention $L = 0 \to R = \infty$.
\end{proposition}
\begin{proof}
  Consider the power series $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$.
  For some fixed $x$, we have:
  \[
    \lim_{n \to \infty} \abs{\frac{c_{n + 1}(x - a)^{n + 1}}{c_n(x - a)^{n}}} = |x - a|\lim_{n \to \infty} \abs{\frac{c_{n + 1}}{c_n}} = |x - a|L
  \]
  So by the root test (\cref{ratioTest}), if $|x - a|L < 1$, then the power series converges absolutely and if $|x - a|L > 1$, the power series diverges by the $n$-th term test (\cref{nthTermTest}).

  In the case of $L = 0$, it must always converge so $R = \infty$, otherwise, it converges absolutely for $|x - a| < \frac{1}{L}$ and diverges for $|x - a| > \frac{1}{L}$ so its radius of convergence must be $R = \frac{1}{L}$.
\end{proof}
\begin{remark}[Note]
  If the limits do not exist but instead converge to infinity, i.e. $\abs{\frac{c_{n + 1}}{c_n}} \to \infty$ or $\sqrt[n]{c_n} \to \infty$, then $R = 0$.
\end{remark}
\begin{example} \label{rConvExamples}
  \begin{enumerate}
    \item The power series $\sum_{n = 0}^{\infty} \frac{x^{n}}{n!}$ converges on all of $\C$, i.e. $R = \infty$.
      This has $c_n = \frac{1}{n!}$.
      Using the ratio test:
      \[
        \lim_{n \to \infty} \frac{\frac{1}{(n + 1)!}}{\frac{1}{n!}} = \lim_{n \to \infty} \frac{1}{n + 1} = 0
      \]
      Therefore $L = 0 \implies R = \infty$.
    \item The power series $\sum_{n = 0}^{\infty} n! x^{n}$ diverges on $\C \setminus \{0\}$ as:
      \[
        \lim_{n \to \infty} \frac{(n + 1)!}{n!} = \lim_{n \to \infty} n = \infty
      \]
      So $L = \infty$ and therefore $R = 0$, so it only converges at its centre $a = 0$.
    \item The power series $\sum_{n = 1}^{\infty} \frac{x^{n}}{n^2}$ converges absolutely on $|x| < 1$ and diverges if $|x| > 1$.
      We can see this using the ratio test:
      \[
        \lim_{n \to \infty} \frac{\frac{1}{(n + 1)^2}}{\frac{1}{n^2}} = \lim_{n \to \infty} \left(\frac{n}{n + 1}\right)^2= 1 \implies R = 1
      \]

      But what about $|x| = 1$?
      For $|x| = 1$, we have:
      \[
        \sum_{n = 1}^{\infty} \abs{\frac{x^{n}}{n^2}} = \sum_{n = 1}^{\infty} \frac{1}{n^2} < \infty
      \]
      So the power series also converges absolutely on $|x| = 1$.
    \item Consider the power series $\sum_{n = 1}^{\infty} \frac{x^{n}}{n}$.
      By using the ratio test as above $R = 1$ so we must have absolute convergence on $|x| < 1$ and divergence on $|x| > 1$, but we cannot make the same conclusion about the behaviour on $|x| = 1$.

      If $x = 1$, then we have divergence as this is just the harmonic series, but at $x = -1$, we have convergence by the alternating series test.
  \end{enumerate}
\end{example}
\subsection{Term by Term Operations}
The functions in the power series are well behaved polynomials $f_n(x) = c_n(x - a)^{n}$ and are hence continuous, integrable, and differentiable.
But can we say something about the continuity integrability and differentiability of the power series:
\[
  f: Y \to \C,\ x \mapsto \sum_{n = 0}^{\infty} c_n(x - a)^{n}
\]
where $Y$ is the set of points for which the power series converges.

Further, can we carry out differentiation and integration term by term?
\begin{align*}
  \deriv{}{x}f(x) &\stackrel{?}= \sum_{n = 0}^{\infty} c_n \deriv{}{x}(x - a)^{n} \\
  \int f(x) \d{x} &\stackrel{?}= \sum_{n = 0}^{\infty} c_n \int (x - a)^{n} \d{x}
\end{align*}
\begin{example}[]
  \begin{enumerate}
    \item Consider again the series $\sum_{n = 0}^{\infty} \frac{x^{n}}{n!}$ from \cref{rConvExamples}.
      We know that this converges on all of $\C$.

      If we try differentiating term by term:
      \[
        \sum_{n = 0}^{\infty} \frac{1}{n!} \deriv{}{x}(x^{n}) = \sum_{n = 1}^{\infty} \frac{x^{n - 1}}{(n - 1)!} = \sum_{n = 0}^{\infty} \frac{x^{n}}{n!}
      \]
      which is the same series so converges on all of $\C$.
      It therefore seems plausible that this could be the same as the derivative of the power series.
    \item Consider again $\sum_{n = 1}^{\infty} \frac{x^{n}}{n^2}$.
      We know that this converges on $D = \{x \in \C: |x| \leq 1\}$.
      \[
        \sum_{n = 1}^{\infty} \frac{1}{n^2} \deriv{}{x}(x^{n}) = \sum_{n = 1}^{\infty} \frac{x^{n - 1}}{n}
      \]
      However, we know this does not converge on all of $D$, for example, at $x = 1$, but does converge on all of $\{x \in \C : |x| < 1\}$.
      So term by term differentiation definitely \textbf{cannot hold on all} of the points of convergence.
      However, we can show that it does \textbf{within} the radius of convergence.
  \end{enumerate}
\end{example}
We first need to check that power series are continuous inside their radius of convergence otherwise there is no hope that we can differentiate them.
\begin{proposition}[Continuity of Power Series]
  If $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ is a power series with a radius of convergence $R > 0$, then the function: \label{powerSeriesCty}
  \[
    f: B_R(a) \to \C,\ X \mapsto \sum_{n = 0}^{\infty} c_n(x - a)^{n} \text{ where } B_R(a) = \{x \in \C: |x - a| < R\}
  \]
  is continuous inside $D_r(a) = \{x \in \C: |z - a| \leq r\}$, for every $r < R$.
\end{proposition}
\begin{remark}
  In the special case of functions of real inputs, we would have $f: (a - R, a + R) \to \C$ is continuous inside $[a - r,  a + r]$ for every $r < R$.
\end{remark}
\begin{proof}
  \nonexaminable
  We can split the power series in to a polynomial of degree $N$, $S_N(x)$, and a tail $T_N(x)$:
  \[
    S_N(x) = \sum_{n \leq N} c_n(x - a)^{n} \text{ and } T_N(x) = \sum_{n \geq N + 1} c_n(x - a)^{n}
  \]
  Let $r \geq 0$ be arbitrary such that $r < R$.
  We would like to bound the tail when we are inside $D_r(a)$:
  \begin{align*}
    |T_N(x)| &\leq \sum_{n \geq N + 1}^{\infty} |c_n||x - a|^{n} \text{ using the triangle-inequality} \\
             &\leq \sum_{n \geq N + 1}^{\infty} |c_n|r^{n} \quad \forall x \in D_r(a)
  \end{align*}
  Since $r < R$, by definition of the radius of convergence, the series $\sum_{n = 0}^{\infty} |c_n|r^{n}$ converges so $\forall \varepsilon > 0, \exists N = N(\varepsilon)$ such that:
  \[
    \sum_{n \geq N + 1}|c_n|r^{n} < \varepsilon
  \]
  For such $N$, we then have:
  \[
     |T_N(x)| < \varepsilon\ \forall x \in D_r(a)
  \]

  Now that we can bound the tail of the power series, take an arbitrary point $x_0 \in D_r(a)$, we want to show that $f$ is continuous here.
  \begin{align*}
    |f(x) - f(x_0)| &= |S_N(x) + T_N(x) - S_N(x_0) - T_N(x_0)| \\
                    &\leq |S_N(x) - S_N(x_0)| + |T_N(x)| + |T_N(x_0)| \\
                    &\leq |S_N(x) - S_N(x)| + 2\varepsilon \quad \forall x \in D_r(a)
  \end{align*}
  Although it seems like a bad idea to make such a harsh approximation using the triangle inequality, in this case, we already knew that the tail can be made arbitrarily small.

  Finally, since $S_N(x)$ is a finite degree $N$ polynomial, it is continuous at $x_0$ so $\exists \delta(\varepsilon)$ such that $|x - x_0| < \delta \implies |S_N(x) - S_N(x_0)| < \varepsilon$, so $f$ is continuous at $x_0$.

  Therefore, given arbitrary $r < R$, $f$ is continuous at $x_0$ for all $x_0 \in D$.
\end{proof}
\begin{remark}[Strategy]
  In this proof, we were able to conclude that for sufficiently large $N$, we can make the tail arbitrarily small and so, in some sense, we can just consider the finite polynomial $S_N$, which is continuous.
\end{remark}
We will first show that power series allow for term by term integration as we can then use the fundamental theorem of calculus to conclude that they are differentiable.
\begin{proposition}[Integration of Power Series]
  If $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ is a real power series with a radius of convergence $R > 0$ then: \label{powerSeriesInt}
  \[
    f: (a - R, a + R) \to \R,\ x \mapsto \sum_{n = 0}^{\infty} c_n(x - a)^{n}
  \]
  is integrable on every $[a - r, a + r]$ for every $r < R$ and:
  \[
    \int_{a}^{x} f(t) \d{x} = \sum_{n = 0}^{\infty} c_n \frac{(x - a)^{n + 1}}{n + 1}
  \]
  That is, integration can be carried out term by term.
\end{proposition}
\begin{proof}
  \nonexaminable
  Let $r \geq 0$ be arbitrary such that $r < R$.
  By \cref{powerSeriesCty}, $f$ is continuous on $[a - r, a + r]$ and so by \cref{continuousIntegrable}, it must also be integrable on $[a - r, a + r]$.

  To show that the integration can be carried out term by term, we can employ a similar strategy as we did for continuity so let $S_N(x)$ and $T_N(x)$ be defined as before.

  For $x \in [a - r, a + r]$, the difference between the integral and the start of the integrated series can be bounded by:
  \begin{align*}
    \abs{\int_{a}^{x} f(t) \d{t} - \sum_{n \leq N} \frac{c_n(x - a)^{n + 1}}{n + 1}} &= \abs{\int_{a}^{x} f(t) \d{t} - \int_{a}^{x} S_N(t) \d{t}} \\
    &= \abs{\int_{a}^{x} T_N(t) \d{t}} \\
    &\leq \sup_{|t - a| < r} |T_N(t)| \abs{\int_{a}^{x} \d{t}} \\
    &\leq |x - a| \cdot \sup_{|t - a| < r} |T_N(t)| \\
    &\leq r \cdot \sup_{|t - a| < r} |T_N(t)| \\
    &\leq r \varepsilon
  \end{align*}
  Where the last line comes from the same argument as in the proof for continuity, where, for any $\varepsilon > 0$, if we choose $N = N(\varepsilon)$ to be sufficiently large, then $\sup_{|t - a| < r} |T_N(t)| < \varepsilon$.

  Therefore:
  \[
    \int_{a}^{x} f(t) \d{t} = \lim_{N \to \infty} \sum_{n = 0}^{N} \frac{c_n(x - a)^{n + 1}}{n + 1} = \sum_{n = 0}^{\infty} \frac{c_n(x - a)^{n + 1}}{n + 1}
  \]
\end{proof}
\begin{proposition}[Differentiation of Power Series]
  If $\sum_{n = 0}^{\infty} c_n(x - a)^{n}$ is a power series with a radius of convergence $R > 0$ then:
  \[
    f: (a - R, a + R) \to \R,\ x \mapsto \sum_{n = 0}^{\infty} c_n(x - a)^{n}
  \]
  is differentiable on every $[a - r, a + r]$ for every $r < R$ with derivative:
  \[
    f'(x) = \sum_{n = 1}^{\infty} nc_n(x - a)^{n - 1}
  \]
  That is, differentiation can be carried out term by term.
\end{proposition}
\begin{proof}
  \nonexaminable
  Let $g(x) = \sum_{n = 1}^{\infty} nc_n(x - a)^{n - 1}$.
  Our strategy here is to show that this series has a radius of convergence of at least $R$ and then employ term by term integration, followed by the fundamental theorem of calculus to show that $f'(x) = g(x)$.

  Let $r \geq 0$ be arbitrary such that $r < R$.
  Since $r < R$, $\exists s \in (r, R)$.
  We can then use this to bound the general sequence term in $g(x)$:
  \begin{align*}
    |n c_n(x - a)^{n - 1}| &= |c_n|s^{n} \cdot \frac{n}{s}\left(\frac{|x - a|}{s}\right)^{n - 1} \\
                           &\leq |c_n|s^{n} \cdot \frac{n}{s}\left(\frac{r}{s}\right)^{n - 1}\ \forall x \text{ s.t. } |x - a| \leq r
  \end{align*}
  Since $s < R$, the series $\sum_{n=0}^{\infty} |c_n|s^{n}$ converges, so by the $n$-th term test (\cref{nthTermTest}), $|c_n|s^{n} \to 0$.
  As $|c_n|s^{n}$ converges, it must be bounded so $\exists M$ such that $|c_n|s^{n} \leq M\ \forall n \geq 0$.
  Therefore, for all $n \geq 0$, the general term of $g$ is bounded by:
  \[
    |n c_n(x - a)^{n - 1}| \leq M \cdot \frac{n}{s}\left(\frac{r}{s}\right)^{n - 1}\ \forall x \text{ s.t. } |x - a| \leq r
  \]
  So:
  \[
    \sum_{n = 1}^{\infty} |nc_n (x - 1)^{n - 1}| \leq \sum_{n = 0}^{\infty} M \cdot \frac{n}{s}\left(\frac{r}{s}\right)^{n - 1}\ \forall x \text{ s.t. } |x - a| \leq r
  \]
  The right hand side of this converges since $\frac{r}{s} < 1$ and so the power series for $g$ is absolutely convergent $\forall x \text{ s.t. } |x - a| \leq r$.
  Since $r < R$ was arbitrary, the radius of convergence of the power series $g(x)$ must be at least $R$.

  Again, for arbitrary $r < R$, since we now know that $g(x)$ has radius of convergence of at least $R$, by \cref{powerSeriesCty}, $g$ is continuous on $[a - r, a + r]$ and further, by \cref{powerSeriesInt}, its integral can be carried out term by term:
  \begin{align*}
    \int_{a}^{x} g(t) \d{t} &= \sum_{n = 1}^{\infty} \left[\int_{a}^{x} nc_n(t - a)^{n - 1} \d{t}\right] \\
                            &= \sum_{n = 1}^{\infty} c_n(x - a)^{n} = f(x) - c_0\ \forall x \in [a - r, a + r]
  \end{align*}
  Since $g$ is continuous on $[a - r, a + r]$, by the FTC (\cref{FTCDiff}):
  \[
    \deriv{}{x}\left[\int_{a}^{x} g(t) \d{t}\right] = g(x)\ \forall x \in [a - r, a + r]
  \]
  so $f$ is differentiable with derivative $f'(x) = g(x)\ \forall x \in [a - r, a + r]$.
  Since $r < R$ was arbitrary, the result follows.
\end{proof}
\begin{remark}
  Since the result of differentiating a power series with radius of convergence $R$ is also a power series with radius of convergence at least $R$, we can iterate this process to conclude that power series are infinitely differentiable within their radius of convergence.
\end{remark}
\begin{example}
  \begin{enumerate}
    \item Consider again $f(x) = \sum_{n = 0}^{\infty} \frac{x^{n}}{n!}$, we know this has $R = \infty$ so it is differentiable everywhere in $\C$ and its derivative is given by term by term differentiation:
      \begin{align*}
        f'(x) &= \deriv{}{x} \sum_{n = 0}^{\infty} \frac{x^{n}}{n!} \\
              &= \sum_{n = 0}^{\infty} \frac{1}{n!}\deriv{}{x}(x^{n}) \\
              &= \sum_{n = 0}^{\infty} \frac{x^{n}}{n!} = f(x)
      \end{align*}
      similarly:
      \begin{align*}
        \int_{0}^{x} f(t) \d{t} &= \int_{0}^{x} \sum_{n = 0}^{\infty} \frac{t^{n}}{n!} \d{t} \\
                                &= \sum_{n = 0}^{\infty} \frac{x^{n + 1}}{(n + 1)!} \\
                                &= \sum_{n = 1}^{\infty} \frac{x^{n}}{n!} \\
                                &= f(x) - 1 = f(x) - f(0)
      \end{align*}
      In the next section, we will use this power series to define the \textit{exponential function}.
      As we have shown, it has the useful property that $f(x) = f'(x)$.
    \item We know that the power series $g(x) = \sum_{n = 1}^{\infty} \frac{x^{n}}{n^2}$ has a radius of convergence $R = 1$.

      So by the above, for $|x| < 1$, we have:
      \[
        g'(x) = \sum_{n = 1}^{\infty} \deriv{}{x}\left(\frac{x^{n}}{n^2}\right) = \sum_{n = 1}^{\infty} \frac{x^{n - 1}}{n}
      \]
      and:
      \[
        \int_{0}^{x} g(t) \d{t} = \sum_{n = 1}^{\infty} \left[\int_{0}^{x} \frac{t^{n}}{n^2} \d{t}\right] = \sum_{n = 1}^{\infty} \frac{x^{n + 1}}{n^2(n + 1)}
      \]
  \end{enumerate}
\end{example}
