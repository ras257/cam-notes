\subsection{Defining the Integral}
We want to define $\int_{a}^{b} f(x) \d{x}$ as the signed area enclosed by $f$ between $a$ and $b$.
This is easy if the region enclosed by the graph is a nice geometric shape, if not, we can try approximating the area using rectangles.
Two ways we could do this are using the left and right Riemann sums:

We first need to define how we can split up the real line between the \textit{integration bounds $a$ and $b$}:
\begin{definition}[Partition/Dissection]
  A \textit{partition} $\mathcal{P}$ or \textit{dissection} of an interval $[a, b]$ is a \textbf{finite subset} of $[a, b]$ with the requirement that $a, b \in \mathcal{P}$.

  We often write $\mathcal{P} = \{x_0, \ldots, x_n\}$ where $a = x_0 < x_1 < \cdots < x_n = b$ and $x_i \in [a, b]\ \forall i$.
\end{definition}
Instead of the left and right Riemann sums, we will instead use the \textit{upper and lower} Riemann sums.
These use the supremum/infimum respectively of $f$ over each interval between the points of a partition.
\begin{definition}[Upper/Lower Riemman Sums]
  For a bounded $f: [a, b] \to \R$ and partition $\mathcal{P}$ of $[a, b]$, we define the \textit{Lower Riemann sum} as:
  \[
    L(f, \mathcal{P}) = \sum_{j = 1}^{n} (x_j - x_{j - 1}) \inf_{x \in I_j} f(x)
  \]
  and the \textit{Upper Riemann sum} as:
  \[
    U(f, \mathcal{P}) = \sum_{j = 1}^{\infty} (x_j - x_{j - 1}) \sup_{x \in I_j} f(x)
  \]
  where $I_j = [x_{j - 1}, x_{j}]$.
\end{definition}
\begin{example}
  Consider again the \textit{Dirichlet function} from \cref{dirichletFunction}:
  \label{dirichletUpperLower}
  \[
    f(x) = 1_\Q(x) = \begin{cases}
    1 & \text{ if } x \in \Q \\
    0 & \text{ if } x \notin \Q
    \end{cases}
  \]
  No matter what $\mathcal{P}$ is, since $\Q$ is dense in $\R$, $I_j = [x_{j - 1}, x_j]$ will always contain a rational and an irrational number, hence:
  \[
    \sup_{x \in I_j} f(x) = 1 \text{ and } \inf_{x \in I_j} f(x) = 1
  \]
  and so:
  \begin{align*}
    U(f, \mathcal{P}) &= 1 \cdot \sum_{j = 1}^{n} (x_j - x_{j - 1}) \\
                       &= x_1 - x_0 + x_2 - x_1 + \cdots + x_n - x_{n - 1} \\
                       &= x_n - x_0 = b - a = 1 - 0 = 1 \\
    L(f, \mathcal{P}) &= 0 \cdot \sum_{j = 1}^{n} (x_j - x_{j - 1}) = 0
  \end{align*}
\end{example}
For bounded $f$, $\exists k \text{ s.t. } \sup_{x \in [a, b]} |f(x)| = k$.
Since $\sup_{x \in I_j} f(x) \leq \sup_{x \in [a, b]} f(x)$, we can bound $U(f, \mathcal{P})$ as:
\begin{align*}
  U(f, \mathcal{P}) &\leq \sum_{j = 1}^{n} (x_{j} - x_{j - 1}) \sup_{x \in [a, b]} |f(x)|  \\
                     &\leq K \sum_{j = 1}^{n} (x_j - x_{j - 1}) = k(b - a)
\end{align*}
and since $\inf_{x \in I_j} f(x) \geq \inf_{x \in [a, b]} f(x) \geq -\sup_{x \in [a, b]} |f(x)| = -k$
\begin{align*}
  L(f, \mathcal{P}) &\geq \sum_{j = 1}^{n} (x_{j} - x_{j -1}) \left[- \sup_{x \in [a, b]} |f(x)|\right] \\
                     &= -k(b - a)
\end{align*}
From the definition, we clearly have that $U(f, \mathcal{P}) \geq L(f, \mathcal{P})$ and so:
\[
  -k(b - a) \leq L(f, \mathcal{P}) \leq U(f, \mathcal{P}) \leq k(b - a)
\]
Geometrically, this is saying that $L$ and $U$ are bounded below by a rectangle from $a$ to $b$ of height $-k$ and above by a rectangle from $a$ to $b$ of height $k$ where $k = \sup_{x \in [a, b]} f(x)$.

Therefore, the sets:
\[
  \{L(f, \mathcal{P}): \mathcal{P} \text{ is a partition of $[a, b]$}\} \text{ and } \{U(f, \mathcal{P}): \mathcal{P} \text{ is a partition of $[a, b]$}\}
\]
are both bounded.
They are also non empty since as $\{a,b\}$ is always a partition of $[a, b]$.
This means that they both have supremum and infimum.
\begin{definition}[Upper/Lower Integral]
  Let $f: [a, b] \to \R$ be a bounded function.
  We define the \textit{lower Riemann integral} of $f$ in $[a, b]$ as:
  \[
    I_{*}(f) = \sup_{\mathcal{P}} L(f, \mathcal{P})
  \]
  and the \textit{upper Riemann integral} of $f$ in $[a, b]$ as:
  \[
    I^{*}(f) = \inf_{\mathcal{P}} U(f, \mathcal{P})
  \]
\end{definition}
\begin{definition}[Riemman Integrable]
  We say that bounded function $f: [a, b] \to \R$ is \textit{Riemann integrable} if $I_{*}(f) = I^{*}(f)$ and then define:
  \label{integralDef}
  \[
    I(f) = \int_{a}^{b} f(x) \d{x} = I_{*}(f) = I^{*}(f)
  \]
  to be the \textit{Riemann integral of $f$ from $a$ to $b$}.
\end{definition}
\begin{remark}[Intuition]
  This is like saying the largest of the pessimistic approximations agrees with the smallest of the optimistic approximations.
\end{remark}
\begin{remark}
  By convention, if we change the order of the bounds, then the sign of the integral flips, that is:
  \[
    \int_{a}^{b} f(x) \d{x} = - \int_{b}^{a} f(x) \d{x}
  \]
\end{remark}
\begin{example}
  \label{dirichletIntegrable}
  Consider again the Dirichlet indicator function from \cref{dirichletUpperLower}.
  Its upper and lower integrals are $I^{*}(f) = 1$ and $I_{*}(f) = 0$, so $f$ is not integrable.
\end{example}
\subsection{Properties of Partitions}
Intuition suggests that a finer partition leads to a better estimate for the area, this turns out to be true:
\begin{lemma}
  If $\mathcal{P}$ and $\mathcal{P}'$ be partitions of $[a, b]$ such that $\mathcal{P} \subseteq \mathcal{P}'$, that is, $\mathcal{P}'$ is a \textit{refinement of $\mathcal{P}$}, then:
  \label{refinementInequality}
  \[
    L(f, \mathcal{P}) \leq L(f, \mathcal{P'}) \leq U(f, \mathcal{P}') \leq U(f, \mathcal{P})
  \]
  for every bounded $f: [a, b] \to \R$.
\end{lemma}
\begin{proof}
  For now, assume that $\mathcal{P}$ and $\mathcal{P}'$ differ by one point only.
  Take $\mathcal{P} = \{x_0, \ldots, x_n\}$.
  Since $\mathcal{P}'$ differs by one point, $\mathcal{P}' = \mathcal{P} \cup \{y\}$ where $y \in (x_{k - 1}, x_k)$ for some $k \in \{1, \ldots, n\}$.
  In this case, we then have:
  \[
    \sup_{x \in [x_{k - 1}, y]} f(x),\ \sup_{x \in [y, x_k]} f(x) \leq \sup_{x \in [x_{k - 1}, x_k]} f(x)
  \]
  and so:
  \begin{align*}
    (y - x_{k - 1})\sup_{x \in [x_{k - 1}, y]} f(x) + (x_k - y) \sup_{x \in [y, x_k]} f(x) &\leq (y - x_{k - 1}+ x_k - y) \sup_{x \in [x_{k - 1}, x_k]} f(x) \\
                                                                                           &= (x_k - x_{k - 1}) \sup_{x \in [x_{k - 1}, x_k]}  f(x)
  \end{align*}
  The sum for $U(f, \mathcal{P}')$ has the terms on the left hand side of the inequality, whereas the sum for $U(f, \mathcal{P})$ instead has the term on the right hand side of the inequality, otherwise, the sums are the same as $\mathcal{P}$ and $\mathcal{P}'$ differ only by one point.
  Therefore, $U(f, \mathcal{P}') \leq U(f, \mathcal{P})$.

  Similarly, using the infimum in the first line instead yields:
  \[
    \inf_{x \in [x_{k - 1}, y]} f(x),\ \inf_{x \in [y, x_k]} f(x) \geq \inf_{x \in [x_{k - 1}, x_k]} f(x)
  \]
  So using a similar argument, we get that $L(f, \mathcal{P}') \geq L(f, \mathcal{P})$.
  Therefore, if $\mathcal{P}'$ and $\mathcal{P}$ differ by one point, the result holds since the middle inequality was shown previously.

  Since partitions are finite subsets, $\mathcal{P}'$ and $\mathcal{P}$ can differ only by a finite number of points.
  So if $\mathcal{P}' \setminus \mathcal{P} = \{y_1, \ldots, y_n\}$, then we can just repeat this argument $n$ times to yield the result.
\end{proof}
\begin{corollary}
  If $\mathcal{P}_1$ and $\mathcal{P}_2$ are partitions of $[a, b]$, then:
  \label{partitionRelation}
  \[
    L(f, \mathcal{P}_1) \leq U(f, \mathcal{P}_2)
  \]
  for every bounded $f: [a, b] \to \R$.
\end{corollary}
\begin{remark}[Note]
  Neither of $\mathcal{P}_1$ and $\mathcal{P}_2$ have to be a refinement of the other for this to hold.
\end{remark}
\begin{proof}
  Since $\mathcal{P}_1, \mathcal{P}_2 \subseteq \mathcal{P}_1 \cup \mathcal{P}_2$, $\mathcal{P}_1$ and $\mathcal{P}_2$ are both refinements of $\mathcal{P}_1 \cup \mathcal{P}_2$.
  Applying \cref{refinementInequality} twice, we have:
  \[
    L(f, \mathcal{P}_1) \leq U(f, \mathcal{P}_1 \cup \mathcal{P}_2) \leq U(f, \mathcal{P}_2)
  \]
  which is the desired result.
\end{proof}
\begin{corollary}
  If $f: [a, b] \to \R$ is bounded, then $I_*(f) \leq I^{*}(f)$.
  \label{upperLowerIntegralRelation}
\end{corollary}
\begin{proof}
  By \cref{partitionRelation}, if we fix some partition $\mathcal{P}$, then $L(f, \mathcal{P}) \leq U(f, \mathcal{P}')$ for all partitions $\mathcal{P}'$.
  Therefore:
  \[
    L(f, \mathcal{P}) \leq \inf_{\mathcal{P}'} U(f, \mathcal{P}')
  \]
  Since $\mathcal{P}$ was arbitrary, this relation is true for all partitions $\mathcal{P}$ and so
  \begin{align*}
    \sup_{\mathcal{P}} L(f, \mathcal{P}) &\leq \inf_{\mathcal{P}'} U(f, \mathcal{P'}) \\
    \implies I_{*}(f) &\leq I^{*}(f)
  \end{align*}
\end{proof}
\begin{remark}
  This means that $f$ is integrable if and only if $I_{*}(f) \geq I^{*}(f)$ since we always have $I_{*}(f) \leq I^{*}(f)$ for bounded $f$.
\end{remark}
\section{Integrability of General Functions}
\subsection{Integrability Criterion}
Is there another definition for Riemann integrability that is more akin of the others we have seen so far in the course such as those for continuity and limits?

We would like to show that for any threshold, we can always find a partition of $[a, b]$ so that the difference between the upper and lower Riemann sums is below the threshold.
\begin{proposition}[Riemman Integrability Critereon]
  Let $f: [a, b] \to \R$ be bounded.

  $f$ is Riemann integrable if and only if:
  \label{integrabilityCritereon}
  \[
    \forall \varepsilon > 0\ \exists \mathcal{P} = \mathcal{P}(\varepsilon) \text{ a partition of } [a, b] \text{ s.t. } U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon
  \]
\end{proposition}
\begin{proof}
  \begin{proofdirection}{Assume $f$ is Riemann integrable}
    From the definition of supremum and infimum, for any $\varepsilon > 0$, we can always find $\mathcal{P}$ and $\mathcal{P}'$ such that:
    \[
      U(f, \mathcal{P}) \leq I^{*}(f) + \frac{\varepsilon}{2} \text{ and } L(f, \mathcal{P}') \geq I_{*}(f) - \frac{\varepsilon}{2}
    \]
    Taking the difference:
    \[
      U(f, \mathcal{P}) - L(f, \mathcal{P'}) \leq \varepsilon
    \]
    To conclude, we need to find a single partition where this difference is smaller than $\varepsilon$.
    Let $\mathcal{P}'' = \mathcal{P} \cup \mathcal{P}'$.
    We see that $\mathcal{P} \subseteq \mathcal{P}$ so $\mathcal{P}''$ is a refinement of $\mathcal{P}$, therefore, by \cref{refinementInequality}, $U(f, \mathcal{P}'') \leq U(f, \mathcal{P})$.
    Similarly, $\mathcal{P}''$ is a refinement of $\mathcal{P}'$ and so $L(f, \mathcal{P}'') \geq L(f, \mathcal{P}')$.

    Combining these inequalities, we have:
    \[
      U(f, \mathcal{P}'') - L(f, \mathcal{P}'') \leq U(f, \mathcal{P}) - L(f, \mathcal{P}') \leq \varepsilon
    \]
  \end{proofdirection}
  \begin{proofdirection}{Assume that we can always find such a partition}
    So for any $\varepsilon > 0$, $\exists \mathcal{P}$ such that $U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon$.
    We also know that for any partition $\mathcal{P}$, $U(f, \mathcal{P}) \geq I^{*}(f)$ and $L(f, \mathcal{P}) \leq I_*(f)$.

    Therefore:
    \[
      0 \leq I^{*}(f) - I_{*}(f) \leq U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon
    \]
    Hence:
    \[
      0 \leq I^{*}(f) - I_{*}(f) < \varepsilon
    \]
    Since $\varepsilon$ can be arbitrarily small, $I^{*}(f) = I_{*}(f)$ and so $f$ is Riemann integrable.
  \end{proofdirection}
\end{proof}
\begin{proposition}
  Let $f: [a, b] \to \R$ be bounded.
  $f$ is Riemann integrable if and only if there exists a sequence of partitions $(\mathcal{P}_n)$ of $[a, b]$ such that:
  \label{integrableSequence}
  \[
    U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) \to 0 \text{ as } n \to \infty
  \]
\end{proposition}
\begin{proof}
  \begin{proofdirection}{Assume $f$ is Riemann integrable}
    Let $\varepsilon = \frac{1}{n}$ for some fixed $n \in \N$, by \cref{integrabilityCritereon}, $\exists \mathcal{P}_n$ such that $U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) < \frac{1}{n}$.
    We can use this to construct a sequence of events $(\mathcal{P}_n)$ by carrying out the above process for every $n \in \N$.
    By construction $0 \leq U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) \leq \frac{1}{n}$, so taking $n \to \infty$, we see that $U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) \to 0$.
  \end{proofdirection}
  \begin{proofdirection}{Assume there exists such a sequence of partitions}
    Let $a_n = U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) \geq 0$.
    Since $a_n \to 0$, for any $\varepsilon > 0\ \exists N(\varepsilon)$ such that:
    \[
      U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) = |a_n| < \varepsilon\ \forall n \geq N
    \]
    So defining $\mathcal{P}'(\varepsilon) = \mathcal{P}_{N(\varepsilon)}$:
    \[
      \forall \varepsilon > 0\ \exists \mathcal{P}'(\varepsilon) \text{ s.t. } U(f, \mathcal{P}') - L(f, \mathcal{P}') < \varepsilon
    \]
    and so $f$ is integrable by \cref{integrabilityCritereon}.
  \end{proofdirection}
\end{proof}
\begin{example}[Examples of Integrable Functions]
  \begin{enumerate}
    \item Let $f: [0, 1] \to \R,\ x \mapsto x$ and define $\mathcal{P}_n = \{0, \frac{1}{n}, \ldots, \frac{n - 1}{n}, 1\}$.

      By definition:
      \begin{align*}
        U(f, \mathcal{P}_n) &= \sum_{j = 1}^{n} \left(\frac{j}{n} - \frac{j - 1}{n}\right) \sup_{x \in [\frac{j - 1}{n}, \frac{j}{n}]} x \\
                            &= \frac{1}{n} \sum_{j = 1}^{n} \frac{j}{n} \\
                            &= \frac{1}{n^2} \cdot \frac{1}{2}n(n + 1) = \frac{1}{2}\left(1 + \frac{1}{n}\right)
      \end{align*}
      Similarly:
      \begin{align*}
        L(f, \mathcal{P}_n) &= \frac{1}{n} \sum_{j = 1}^{n} \inf_{[\frac{j - 1}{n}, \frac{j}{n}]} x \\
                            &= \frac{1}{n} \sum_{j = 1}^{n} \frac{(j - 1)}{n} \\
                            &= \frac{1}{n^2} \cdot \frac{1}{2}n(n - 1) = \frac{1}{2}\left(1 - \frac{1}{n}\right)
      \end{align*}

      \textbf{Using Original Definition -- \Cref{integralDef}}\par
      Since the set of $\{\mathcal{P}_n: n \in \N\}$ is a subset of the set of all possible partitions, we have:
      \[
        I_{*}(f) = \sup_{\mathcal{P}} L(f, \mathcal{P}) \geq \sup_{n \in \N} L(f, \mathcal{P}_n) = \frac{1}{2}
      \]
      and:
      \[
        I^{*}(f) = \inf_{\mathcal{P}} U(f, \mathcal{P}) \leq \inf_{n \in \N} U(f, \mathcal{P}_n) = \frac{1}{2}
      \]
      Thus $I^{*}(f) \leq I_{*}(f)$, so by \cref{upperLowerIntegralRelation}, $I_{*}(f) = I^{*}(f)$.
      So $f$ is integrable and $\int_{0}^{1} f(x) \d{x} = \frac{1}{2}$.

      \textbf{Using Riemann Integrability Sequence Criterion -- \cref{integrableSequence}}\par
      Using $(\mathcal{P}_n)$ as our sequence of events, we see that:
      \[
        U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) = \frac{1}{n} \to 0 \text{ as $n \to \infty$}
      \]
      So $f$ is integrable.
    \item Define:
      \[
        f: [0, 1] \to \R,\ x\mapsto \begin{cases}
        0 & \text{ if } x\leq \frac{1}{2} \\
        1 & \text{ if } x > \frac{1}{2}
        \end{cases}
      \]
      Taking $\mathcal{P}_n$ as before for $n \geq 3$, for $k \in \{1, \ldots, n\}$ we have:
      \[
        \sup_{x \in [x_{k - 1}, x_{k}]} f(x) - \inf_{x \in [x_{k - 1}, x_{k}]} f(x) = \begin{cases}
        1 & \text{ if } \frac{1}{2} \in [x_{k - 1}, x_{k}] \\
        0 & \text{otherwise}
        \end{cases}
      \]
      Since $\frac{1}{2}$ can be in exactly one interval of the partition:
      \begin{align*}
        U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) &= \sum_{j = 1}^{n} (x_j - x_{j - 1}) \left[\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right] \\
                                                  &= \sum_{j = 1}^{n} \frac{1}{n} \left[\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right] \\
                                                  &= \frac{1}{n} \cdot 1 \to 0 \text{ as } n \to \infty
      \end{align*}
      So by \cref{integrableSequence}, $f$ is integrable.
  \end{enumerate}
\end{example}
\subsection{Monotone Functions}
It is quite tedious to check if a function is integrable using these methods so it would be useful to know what classes of functions are integrable.
\begin{proposition}[Monotone $\implies$ Integrable]
  If $f: [a, b] \to \R$ is monotone, then it is integrable.
  \label{motoneIntegrable}
\end{proposition}
\begin{proof}
  We need to show that $f$ is bounded for it to be able to be integrable but since $f$ is monotone, $f$ is bounded by $\max\{f(a), f(b)\}$.

  By definition:
  \[
    U(f, \mathcal{P}) -L(f, \mathcal{P}) = \sum_{j = 1}^{n} \left[\sup_{x \in I_{j}} f(x) - \inf_{x \in I_j} f(x)\right]|I_j|
  \]
  where $I_j = [x_{j - 1}, x_j]$ and $|I_j| = x_j - x_{j - 1}$.
  Since $f$ is monotone, the supremum/infimum over an interval $I_j$ are easy to find.
  WLOG assume $f$ to be increasing as the proof is almost identical for decreasing $f$.
  Therefore:
  \[
    \sup_{x \in I_j} f(x) = f(x_j) \text{ and } \inf_{x \in I_j}  f(x) = f(x_{j - 1})
  \]
  Hence:
  \[
    U(f, \mathcal{P}) - L(f, \mathcal{P}) = \sum_{j = 1}^{n} [f(x_j) - f(x_{j - 1})]|I_j|
  \]
  for any partition $\mathcal{P}$ of $[a, b]$.

  Let $\mathcal{P}_n = \{a, a + \frac{b - a}{n}, \ldots, a + \frac{n - 1}{n}(b - a), b\}$ so that $|I_j| = \frac{b - a}{n}$.
  For this partition, we have:
  \begin{align*}
    U(f, \mathcal{P}_n) - L(f, \mathcal{P}_n) &= \frac{b - a}{n} \sum_{j = 1}^{n} \left[f\left(a + j \cdot \frac{b - a}{n}\right) - f\left(a + (j - 1) \cdot\frac{b - a}{n}\right)\right] \\
                                              &= \frac{b - a}{n} (f(b) - f(a)) \text{ by telescoping} \\
                                              &\to 0 \text{ as } n \to \infty
  \end{align*}
  So by \cref{integrableSequence}, $f$ is integrable.
\end{proof}
\subsection{Continuous Functions}
\begin{proposition}[Continious $\implies$ Integrable]
  If $f: [a, b] \to \R$ is continuous, then it is integrable.
  \label{continuousIntegrable}
\end{proposition}
\begin{proof}
  Again, we first need to check that $f$ is bounded.
  Since $f$ is continuous and $[a, b]$ is a closed and bounded set, by extreme value theorem (\cref{EVT}), $f$ is bounded.

  Suppose, for contradiction, that $f$ is not integrable.
  Then by negating \cref{integrabilityCritereon}, $\exists \varepsilon > 0$ such that $\forall \mathcal{P}$ that are partitions of $[a, b]$ we have:
  \[
    U(f, \mathcal{P}) - L(f, \mathcal{P}) \geq \varepsilon
  \]
  Using the definition of the upper and lower sums:
  \begin{align*}
    U(f, \mathcal{P}) - L(f, \mathcal{P}) &= \sum_{j = 1}^{n} |I_j|\left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \\
                                          &\leq \left[\sum_{j = 1}^{n} |I_j|\right] \max\limits_{1 \leq j \leq n}\left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \\
                                          &= (b - a)\max\limits_{1 \leq j \leq n}\left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right)
  \end{align*}
  Therefore:
  \[
    \max\limits_{1 \leq j \leq n}\left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \geq \frac{1}{b - a}(U(f, \mathcal{P}) - L(f, \mathcal{P})) \geq \frac{\varepsilon}{b - a}
  \]
  So $\exists j \in \{1, \ldots, n\}$ such that $\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x) \geq \frac{\varepsilon}{b - a}$.
  By extreme value theorem since $I_j$ is closed and bounded, $\exists y, z \in [x_{j - 1}, x_j]$ such that $f(y) = \sup_{x \in I_j} f(x)$ and $f(z) = \inf_{x \in I_j} f(x)$ and so $f(y) - f(z) \geq \frac{\varepsilon}{b - a}$.

  Since this is true for all partitions, it should also be true for $\mathcal{P}_n$ as defined in the proof for \cref{motoneIntegrable}.
  Repeating the above process for each $n$, we get sequences $(y_n)$ and $(z_n)$ with $|f(y_n) - f(z_n)| \geq \frac{\varepsilon}{b - a}$.
  Furthermore, since $|I_j| = \frac{b - a}{n}\ \forall j$ and $y_n, z_n$ are in the same interval, $|y_n - z_n| \leq \frac{b - a}{n}$.

  We want to use this to conclude that $f$ cannot be sequentially continuous, the problem is that we do not know if $(z_n)$ and $(y_n)$ are convergent sequences.
  However, we do know that both are bounded sequences, so we can apply the Bolzano Weierstrass theorem, and so there is a convergent subsequence $(y_{n_k})$ which converges to $y$.
  Since $(y_{n_k})$ is a sequence on $[a, b]$ which is closed, $y \in [a, b]$.

  We want to show that $z_{n_k}$ convergences to the same limit:
  \[
    |z_{n_k} - y_{n_k}| < \frac{b - a}{n_{k}} \to 0 \text{ as } k \to \infty
  \]
  thus $z_{n_k} \to y$ too.

  Therefore $y_{n_k}, z_{n_k} \to y$ but for some $\varepsilon > 0$:
  \[
    |f(z_{n_k}) - f(y_{n_k})| > \frac{\varepsilon}{b - a}\ \forall k \implies f(z_{n_k}) - f(y_{n_k}) \centernot\to 0
  \]
  Thus $(f(y_{n_k}))_k$ and $(f(y_{n_k}))_k$ cannot have the same limit.
  Hence $f$ cannot be sequentially continuous and so by \cref{sequentialContinuityEquivalence} $f$ is not continuous, which is a contradiction.
  Therefore, $f$ must be integrable, as required.
\end{proof}
\subsection{Piecewise Continuous Functions}
\begin{definition}[Piecewise Continuous]
  We say that a function $f: [a, b] \to \R$ is \textit{piecewise continuous} if there is a partition $\mathcal{P}$ of $[a, b]$ such that for all $j \in \{1, \ldots, n\}$, the function with its domain restricted to $(x_{j - 1}, x_{j})$, i.e $\at{f}{(x_{j - 1}, x_j)}$, satisfies:
  \begin{itemize}
    \item It is continuous.
    \item The limit as $x \to x_j^{-}$ exists and is finite.
    \item The limit as $x \to x_{j - 1}^{+}$ exists and is finite.
  \end{itemize}
\end{definition}
Before showing that piecewise continuous functions are integrable, we need to show that if two functions disagree on a finite set of points, and one is integrable, then both are integrable and their integrals have the same value.
\begin{lemma}
  Let $f, g: [a, b] \to \R$ be bounded functions.
  If $g$ is integrable and the set of points on which $f$ and $g$ disagree is finite, that is:
  \label{integrabilityFiniteDifference}
  \[
    \{x \in [a, b]: f(x) \neq g(x)\} = \{z_1, \ldots, z_N\}
  \]
  for some finite $N$, then $f$ is integrable and:
  \[
    \int_{a}^{b} f(x) \d{x} = \int_{a}^{b} g(x) \d{x}
  \]
\end{lemma}
\begin{proof}\par
  \textbf{Integrability --} Since and $f, g$ are bounded, supremum of their modulus exist for both and so $\exists M = \max\{\sup_{x \in [a, b]} |f(x)|, \sup_{x \in [a, b]} |g(x)|\}$.
  If we fix an $\varepsilon > 0$, then by \cref{integrabilityCritereon} $\exists \mathcal{P}$ a partition of $[a, b]$ such that:
  \[
    U(g, \mathcal{P}) - L(g, \mathcal{P}) < \varepsilon
  \]
  The goal here is to construct a refinement of $\mathcal{P}$ which isolates the problematic points and gives them very little ``weight'' by putting them in arbitrarily small intervals.

  Construct the intervals $J_{k} = [z_k - r_k, z_k + r_k]$ for $r_k > 0$ satisfying:
  \[
    \sum_{k = 1}^{N} |J_k| = 2 \sum_{k = 1}^{N} r_k < \varepsilon
  \]
  Define the refinement $\mathcal{P}'$ of $\mathcal{P}$ as:
  \[
    \mathcal{P}' = \mathcal{P} \cup \{z_1 \pm r_1, \ldots, z_N \pm r_N\}
  \]

  Since $\mathcal{P}'$ is a refinement of $\mathcal{P}$, by \cref{refinementInequality}, we have:
  \[
    U(g, \mathcal{P}') - L(g, \mathcal{P}') \leq U(g, \mathcal{P}) - L(g, \mathcal{P}) < \varepsilon
  \]
  We now need to compare this to $U(f, \mathcal{P}') - L(f, \mathcal{P}')$.
  For convenience, define $J = \bigcup_{k = 1}^{N} J_k$.
  We can first split the sum into two parts, one over the intervals that do not contain any disagreement points ($I_j \centernot\subset J$), and one over the intervals that contain a disagreement point ($I_j \subset J$):
  \begin{align*}
    U(f, \mathcal{P}') - L(f, \mathcal{P}') &= \sum_{j = 1}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \\
                                            &= \sum_{\substack{j = 1\\I_j \subset J}}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \\&+ \sum_{\substack{j = 1\\I_j \centernot\subset J}}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right)
  \end{align*}
  If $I_j \centernot\subset J$, then $f = g$ everywhere in $I_j$, and so:
  \begin{align*}
    \sum_{\substack{j = 1\\I_j \centernot\subset J}}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) &= \sum_{\substack{j = 1\\I_j \centernot\subset J}}^{n} |I_j| \left(\sup_{x \in I_j} g(x) - \inf_{x \in I_j} g(x)\right) \\
                                                                                                                          &\leq \sum_{j = 1}^{n} |I_j| \left(\sup_{x \in I_j} g(x) - \inf_{x \in I_j} g(x)\right) \\
                                                                                                                          &= U(g, \mathcal{P}') - L(g, \mathcal{P}') < \varepsilon
  \end{align*}

  If $I_j \subset J$, then the only thing we know about $f$ is that it is bounded and so:
  \[
    \sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x) \leq 2M
  \]
  Hence:
  \[
    \sum_{\substack{j = 1\\I_j \subset J}}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \inf_{x \in I_j} f(x)\right) \leq 2 M \sum_{\substack{j = 1\\I_j \subset J}}^{n} |I_j| \leq 2M \cdot \sum_{k = 1}^{N} |J_k| < 2M \varepsilon
  \]
  since there are $N$ intervals such that $I_j \subset J$ and for each of these, their length must be less than the corresponding $J_k$.
  Thus for any $\varepsilon > 0$, $\exists \mathcal{P}'$ such that $U(f, \mathcal{P}') - L(f, \mathcal{P}') \leq (2M + 1)\varepsilon$ and so $f$ is integrable.

  \textbf{Equality of Integrals --} Finally, since $f$ is integrable also $\int_{a}^{b} f(x) \d{x}$ is well defined and so $L(f, \mathcal{P}') \leq \int_{a}^{b} f(x) \d{x} \leq U(f, \mathcal{P}')$ and $L(g, \mathcal{P}') \leq \int_{a}^{b} g(x) \d{x} \leq U(g, \mathcal{P}')$.

  We also have that that:
  \begin{align*}
    U(f, \mathcal{P}') - U(g, \mathcal{P}') \leq \sum_{\substack{j = 1 \\ I_j \subset J}}^{n} |I_j| \left(\sup_{x \in I_j} f(x) - \sup_{x \in I_j} g(x)\right) \leq 2M\sum_{\substack{j = 1 \\ I_j \subset J}}^{n} |I_j| < 2M\varepsilon
  \end{align*}
  so:
  \begin{align*}
    \int_{a}^{b} f(x) \d{x} - \int_{a}^{b} g(x) \d{x} &\leq U(f, \mathcal{P}') - L(g, \mathcal{P}') \\
                                                      &= U(f, \mathcal{P}') - U(g, \mathcal{P}') + U(g, \mathcal{P}') - L(g, \mathcal{P}') \\
                                                      &< 2M\varepsilon + \varepsilon = (2M + 1)\varepsilon
  \end{align*}
  and similarly $L(f, \mathcal{P}') - L(g, \mathcal{P}') > -2M\varepsilon$ so $\abs{\int_{a}^{b} f(x) \d{x} - \int_{a}^{b} g(x) \d{x}} < (2M + 1)\varepsilon\ \forall \varepsilon > 0$.
  Thus the two integrals must be equal.
\end{proof}
We also need to show that adding two integrals acts as we would expect.
\begin{proposition}[Additivity of the Integral]
  If $f: [a, b] \to \R$ and $c \in (a, b)$, then $f$ is integrable if and only if $\at{f}{[a, c]}$ and $\at{f}{[c, b]}$ are integrable. \label{integralAdditivity}

  Furthermore:
  \[
    \int_{a}^{b} f(x) \d{x} = \int_{a}^{c} \at{f}{[a, c]}(x) \d{x} + \int_{c}^{b} \at{f}{[c, b]}(x) \d{x}
  \]
\end{proposition}
\begin{proof}
  For convenience, define $I_L = [a, c]$, $I_R = [c, b]$, $f_L = \at{f}{I_L}$, and $f_R = \at{f}{I_R}$.
  \begin{proofdirection}{Assume $f$ is integrable}
    If we fix $\varepsilon > 0$, then by \cref{integrabilityCritereon}, $\exists \mathcal{P}$ of $[a, b]$ such that $U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon$.
    WLOG, we can assume that $c = x_{\ell}$ for some $\ell \in \{1, \ldots, n\}$, as if not, we can just refine $\mathcal{P}$ by adding $c$ which, by \cref{refinementInequality}, would only make $U(f, \mathcal{P}) - L(f, \mathcal{P})$ smaller.
    We can then split $\mathcal{P}$ up as $\mathcal{P} = \mathcal{P}_L \cup \mathcal{P}_R$ where $\mathcal{P}_L = \{x_0, \ldots, x_{\ell}\}$ and $\mathcal{P}_L = \{x_{\ell}, \ldots, x_n\}$.
    Both $\mathcal{P}_L$ and $\mathcal{P}_R$ are valid partitions of $[a, c]$ and $[c, b]$ respectively since they contain the corresponding endpoints.

    We can then use this partitions to split up the upper and lower sums:
    \begin{align*}
      U(f, \mathcal{P}) = U(f_L, \mathcal{P}_L) + U(f_R, \mathcal{P}_R) \\
      L(f, \mathcal{P}) = L(f_L, \mathcal{P}_L) + L(f_R, \mathcal{P}_R)
    \end{align*}
    Using the earlier inequality:
    \[
      U(f_L, \mathcal{P}_L) - L(f_L, \mathcal{P}_L) + U(f_R, \mathcal{P}_R) - L(f_R, \mathcal{P}_R) = U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon
    \]
    We also know that $U(f_L, \mathcal{P}_l) - L(f_L, \mathcal{P}_L) \geq 0$ and $U(f_R, \mathcal{P}_R) - L(f_R, \mathcal{P}_R) \geq 0$ so since their sum is less than $\varepsilon$, both must be individually less than $\varepsilon$ and so $f_L$ and $f_R$ are integrable.
  \end{proofdirection}
  \begin{proofdirection}{Assume $f_L$ and $f_R$ are integrable}
    If we fix $\varepsilon > 0$, then $\exists \mathcal{P}_L$ and $\mathcal{P}_R$ which are partitions of $I_L$ and $I_R$ respectively such that:
    \[
      U(f_R, \mathcal{P}_R) - L(f_R, \mathcal{P}_R) < \varepsilon \text{ and } U(f_L, \mathcal{P}_L) - L(f_L, \mathcal{P}_L) < \varepsilon
    \]
    We can then construct a partition of $[a, b]$ as $\mathcal{P} = \mathcal{P}_L \cup \mathcal{P}_R$.
    Therefore:
    \[
      U(f, \mathcal{P}) - L(f, \mathcal{P}) = U(f_R, \mathcal{P}_R) - L(f_R, \mathcal{P}_R) + U(f_L, \mathcal{P}_L) - L(F_L, \mathcal{P}_L) < 2\varepsilon
    \]
    Thus $f$ is integrable.
  \end{proofdirection}\par
  \textbf{Value of Integral --} We know that $f$ is integrable in $[a, b]$.
  Furthermore:
  \begin{align*}
    \int_{a}^{b} f(x) \d{x} \geq L(f, \mathcal{P}) &= L(f_L, \mathcal{P}_L) + L(f_R, \mathcal{P}_R) \\
                            &> U(F_L, \mathcal{P}_L) - \varepsilon + U(f_R, \mathcal{P}_R) - \varepsilon \\
                            &\geq \int_{a}^{c} f_L(x) \d{x} + \int_{c}^{b} f_R(x) \d{x} - 2\varepsilon
  \end{align*}
  and by a similar argument:
  \[
    \int_{a}^{b} f(x) \d{x} < \int_{a}^{c} f_L(x) \d{x} + \int_{c}^{b} f_R(x) \d{x} + 2\varepsilon
  \]
  Thus, $\abs{\int_{a}^{b} f(x) \d{x} - \left(\int_{a}^{c} f_L(x) \d{x} + \int_{c}^{b} f_R(x) \d{x}\right)} < 2\varepsilon\ \forall \varepsilon > 0$ so the result follows.
\end{proof}
We can now show that piecewise continuous functions are integrable.
\begin{proposition}[Piecewise Continiuous $\implies$ Integrable]
  If $f: [a, b] \to \R$ is a piecewise continuous function, then it is integrable and: \label{piecewiseIntegrable}
  \[
    \int_{a}^{b} f(x) \d{x} = \sum_{j = 1}^{n} \int_{x_{j - 1}}^{x_j} \overline{f}_j(x)  \d{x}
  \]
  where:
  \[
    \overline{f}_j(x) = \begin{cases}
    f(x) & \text{ if } x \in (x_{j - 1}, x_j) \\
    \lim_{x \to x_j^-} f(x) & \text{ if } x = x_j  \\
    \lim_{x \to x_{j - 1}^{+}} f(x) & \text{ if } x=x_{j - 1}
    \end{cases}
  \]
  Note that the limits always exist since $f$ is piecewise continuous.
\end{proposition}
\begin{proof}
  Firstly, by \cref{continuousIntegrable}, we have that $\overline{f}_j$ are integrable.
  We can then use \cref{integrabilityFiniteDifference}, to conclude that $\at{f}{[x_{j - 1}, x_j]}$ are integrable and that:
  \[
    \int_{x_{j - 1}}^{x_j} \overline{f}_j(x) \d{x} = \int_{x_{j - 1}}^{x_j} f(x) \d{x}
  \]
  since $\at{f}{[x_{j - 1}, x_j]}$ and $\overline{f}_j$ can differ only by their values at $x_{j - 1}$ and $x_j$.

  Finally, by repeated use of \cref{integralAdditivity}, we have that:
  \[
    \sum_{j = 1}^{n} \int_{x_{j - 1}}^{x_j} \overline{f}_j(x) \d{x} = \sum_{j = 1}^{n} \int_{x_{j - 1}}^{x_j} f(x) \d{x} = \int_{a}^{b} f(x) \d{x}
  \]
  which is the desired result.
\end{proof}
\subsection{Thomae Function and Discontinuities}
We might wonder if continuous, piecewise continuous and monotone functions the only Riemann integrable functions?
This however, is not the case:
\begin{example}[Thomae Function/Popcorn Function]
  Consider the \textit{Thomae} or \textit{Popcorn} function defined by:
  \[
    f: [0, 1] \to \R,\ x \mapsto \begin{cases}
    \frac{1}{q} & \text{if } x = \frac{p}{q} \text{ with } p, q \in \Z \text{ coprime} \\
    0 & x \in \R \setminus \Q
    \end{cases}
  \]
  Since $\R \setminus \Q$ is dense in $\R$, regardless of the partition, it will always contain a rational and so we will always get $L(f, \mathcal{P}) = 0$, hence $I_*(f) = 0$.
  Now, to show that $f$ is integrable, since $L(f, \mathcal{P}) = 0$ always, we just need to show that $\forall \varepsilon > 0$, $\exists \mathcal{P}$ of $[0, 1]$ such that $U(f, \mathcal{P}) < \varepsilon$.

  For any fixed $\varepsilon > 0$, pick $N \in \N$ such that $N > \frac{1}{\varepsilon}$ and set:
  \[
    X_N = \left\{x \in [0, 1]: f(x) \geq \frac{1}{N}\right\} \subset \left\{\frac{p}{q}: p, q \in \Z,\ 1 \leq q \leq N,\ 0 \leq p \leq q\right\}
  \]
  First, note that $X_n \subset \Q$ as $f$ maps all irrationals to $0$.
  Secondly, the rightmost set is finite and since $X_N$ is a subset of it, $X_N = \{y_1, \ldots, y_M\}$ for some finite $M$.

  We then construct an partition $\mathcal{P}$ such that:
  \begin{itemize}
    \item Each $y_k$ is in some subinterval of $\mathcal{P}$ and no two are in the same subinterval.
    \item Each subinterval containing a $y_k$ has length $\leq \frac{\varepsilon}{M}$.
      This will ensure that the points in $X_N$, i.e. those which are far from $0$, will have little ``weight'' when calculating the upper sum.
  \end{itemize}
  For notational convenience, consider the partition to be a collection of intervals.
  Similarly to in the proof for \cref{integrabilityFiniteDifference}, we can split the upper sum into two parts:
  \[
    U(f, \mathcal{P}) = \sum_{\substack{I \in \mathcal{P} \\ I \cap X_N = \emptyset}} |I| \sup_{x \in I} f(x) + \sum_{\substack{I \in \mathcal{P} \\ I \cap X_N \neq \emptyset}} |I| \sup_{x \in I} f(x) \\
  \]
  If $I \cap X_n = \emptyset$, then $\sup_{x \in I} f(x) \leq \frac{1}{N}$ by construction of $X_N$.
  If $I \cap X_n \neq \emptyset$, then $|I| \leq \frac{\varepsilon}{M}$ by construction of $\mathcal{P}$, and, $\sup_{x \in I} f(x) \leq 1$ as $f(x) \leq 1\ \forall x$.
  Therefore:
  \begin{align*}
    U(f, \mathcal{P}) &\leq \sum_{\substack{I \in \mathcal{P} \\ I \cap X_N = \emptyset}} |I| \cdot \frac{1}{N} + \sum_{\substack{I \in \mathcal{P} \\ I \cap X_N \neq \emptyset}} |I| \\
                      &\leq \frac{1}{N} \cdot 1 + M \cdot \frac{\varepsilon}{M} = \frac{1}{N} + \varepsilon < 2\varepsilon
  \end{align*}
  as there are $M$ subintervals which have nonempty intersection with $X_N$.

  Therefore, $f$ is integrable.
  Although we do not know the value of $U(f, \mathcal{P})$ for any $\mathcal{P}$, because it is integrable $\int_{a}^{b} f(x) \d{x} = I^{*}(f) + I_{*}(f) = 0$.

  This function has a similar form to to the Dirichlet function from \cref{dirichletIntegrable}, yet their integrability properties are completely different.
\end{example}
\begin{proposition}[Countable Discontinuities $\implies$ Integrable]
  Let $f: [a, b] \to \R$ be bounded, and define the set of discontinuity points of $f$ as:
  \[
    \mathcal{D} = \{x \in [a, b]: f \text{ is not continuous at $x$}\}
  \]
  We then have the following:
  \begin{enumerate}
    \item $\mathcal{D}$ is finite $\implies$ $f$ is Riemann integrable.
    \item $\mathcal{D}$ is countable $\implies$ $f$ is Riemann integrable.
  \end{enumerate}
\end{proposition}
\begin{proof}[i]
  See Q13b Example Sheet 3.
\end{proof}
\begin{proof}[2]
  \textbf{Non-Examinable} and so not provided.
\end{proof}

  If $f$ is not Riemann integrable, then $\mathcal{D}$ cannot be countable.
  However, there are functions with uncountable $\mathcal{D}$ that are Riemann integrable.
  For example, the indicator function of the \textit{Cantor set} has this property.
\end{remark}
\section{Properties of Integrals}
To prove some basic properties of integrals, we first need to prove some properties about infimum, supremum, and also upper and lower sums.
\subsection{Properties of Infimum and Supremum}
\begin{lemma}[Properties of Infimum and Supremum]
  Let $I$ be a closed bounded interval and let $f, g: I \to \R$ be bounded functions.
  We then have the following: \label{infSupProperties}
  \begin{enumerate}
    \item If $f(x) \leq g(x)\ \forall x \in I$, then $\sup_{x \in I} f(x) \leq \sup_{x \in I} g(x)$ and $\inf_{x \in I} f(x) \leq \inf_{x \in I} g(x)$.
    \item $\sup_{x \in I} [-f(x)] = - \inf_{x \in I} f(x)$.
    \item For fixed $\lambda > 0$, $\sup_{x \in I} [\lambda f(x)] = \lambda \sup_{x \in I} f(x)$.
    \item $\sup_{x \in I} (f + g) \leq \sup_{x \in I} f(x) + \sup_{x \in I} g(x)$ and $\inf_{x \in I} (f + g) \geq \inf_{x \in I} f(x) + \inf_{x \in I} g(x)$ \item $\sup_{x \in I} |f| - \inf_{x \in I} |f| \leq \sup_{x \in I} f - \inf_{x \in I} f$.
  \item $\sup_{x \in I} f^2 - \inf_{x \in I} f^2 \leq 2 [\sup_{x \in I} |f|][\sup_{x \in I} f(x) - \inf_{x \in I} f(x)]$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  \begin{enumerate}
    \item For all $x \in I$, we have $f(x) \leq g(x) \leq \sup_{y \in I} g(y)$.
      The right hand side of this is just a constant, so taking supremum of both sides with respect to $x$, we have $\sup_{x \in I} f(x) \leq \sup_{y \in X} g(y) = \sup_{x \in I} g(y)$.

      Similarly, $\inf_{y \in I} f(y) \leq f(x) \leq g(x)$ so $\inf_{x \in I} f(x) = \inf_{y \in I} f(y) \leq \inf_{x \in I} g(x)$.
    \item $\inf_{I} f \leq f(x) \iff - \inf_{I} f \geq -f(x) \ \forall x \in I$ and so $-\inf_{I} f \geq \sup_{I} (-f)$.

      Similarly, $-f(x) \leq \sup_{I} (-f) \iff f(x) \geq -\sup_{I} (-f)\ \forall x \in I$ and so $\inf_{I} f \geq -\sup_{I} f \implies -\inf_{I} f \leq \sup_{I} (-f)$.

      Combining the above, we have:
      \[
        -\inf_{I} f \leq \sup_{I} (-f) \leq -\inf_{I} f \implies \sup_{I} (-f) = - \inf_{I} f
      \]
    \item We know that $f(x) \leq \sup_{I} f\ \forall x \iff \lambda f(x) \leq \lambda \sup_{I} f\ \forall x \implies \sup_{I} (\lambda f) \leq \lambda \sup_{I} f$.
      Similarly, $\lambda f(x) \leq \sup_{I} (\lambda f)\ \forall x \in I \iff f(x) \leq \frac{1}{\lambda}\sup_{I} f\ \forall x \in I \implies \sup_{I} f \leq \frac{1}{\lambda}\sup_{I} (\lambda f) \implies \lambda \sup_{I} f \leq \sup_{I} (\lambda f)$.

      Combining the above, we have:
      \[
        \lambda \sup_{I} f \leq \sup_{I} (\lambda f) \leq \lambda \sup_{I} f \implies \sup_{I} (\lambda f) = \lambda \sup_{I} f
      \]
    \item We know that $f(x) \leq \sup_{I} f\ \forall x \in I$ and $g(x) \leq \sup_{I} g\ \forall x \in I$ and so $f(x) + g(x) \leq \sup_{I} f + \sup_{I} g \implies \sup_{I} (f + g) \leq \sup_{I} f + \sup_{I} g$.

      Using the above result and \textbf{(ii)}, we have:
      \[
        \inf_{I} (f + g) = - \sup_{I} (-f - g) \geq -(\sup_{I} (-f) + \sup_{I} (-g)) = \inf_{I} f + \inf_{I} g
      \]
    \item
      \begin{proofcases}
        \begin{case}{$f \geq 0$ or $f \leq 0$ on all of $I$}
          If $f \geq 0$ on all of $I$, then $|f| = f$ so we have equality

          If $f \leq 0$ on all of $I$, then $|f| = -f$ so by \textbf{(ii)}:
          \[
            \sup_{I} |f| - \inf_{I} |f| = \sup_{I} (-f) - \inf_{I} (-f) = - \inf_{I} f + \sup_{I} f
          \]
          So we again have equality.
        \end{case}
        \begin{case}{$\inf_I f < 0 < \sup_I f$}
          We then have that:
          \[
            \sup_{x \in I} |f| - \inf_{I} |f| \leq \sup_{x \in I} |f| \leq \max\left\{\sup_{I} f, \sup_{I} (-f)\right\} \\
          \]
          By \textbf{ii}, $\sup_{I} (-f) = - \inf_{x \in I} f > 0$.
          We also know that $\sup_{I} f > 0$ and so:
          \[
            \sup_{I} |f| - \inf_{I} |f| = \max\left\{\sup_{I} f, -\inf_{I} f\right\} \leq \sup_{I} f - \inf_{I} f
          \]
          as the maximum of two positive numbers is less than their sum.
        \end{case}
      \end{proofcases}
    \item Consider arbitrary $x, y \in I$:
      \begin{align*}
        (f(x))^2 - (f(y))^2 &= (f(x) + f(y))(f(x) - f(y)) \\
                            &\leq 2 \left(\sup_{x \in I} |f|\right)\left(\sup_{x \in I}\sup_{y \in I}[f(x) - f(y)]\right) \\
                            &= 2 \left(\sup_{x \in I} |f|\right)\left(\sup_{x \in I}f(x) - \inf_{x \in I} f(x)\right) \text{ using \textbf{(ii)}}
      \end{align*}
      This inequality is true for all $x, y \in I$ and its right hand side is just a constant so the supremum with respect to $x$ and $y$ on the left hand side must be less than or equal to this.
      Therefore, using \textbf{(ii)} again, we have:
      \[
        \sup_{I} \sup_{I}[(f(x))^2 - (f(y))^2] = \sup_{I} f^2 - \inf_{I} f^2 \leq 2 \left(\sup_{I} |f|\right)\left(\sup_{I}f(x) - \inf_{I} f(x)\right)
      \]
  \end{enumerate}
\end{proof}
\subsection{Properties of Upper and Lower Sums}
\begin{lemma}[Properties of Upper and Lower Sums]
  Let $f, g: [a, b] \to \R$ be bounded functions and let $\mathcal{P}$ be a partition of $[a, b]$.
  We then have the following: \label{upperLowerProperties}
  \begin{enumerate}
    \item If $f(x) \leq g(x)\ \forall x \in [a, b]$, then $L(f, \mathcal{P}) \leq L(g, \mathcal{P})$ and $U(f, \mathcal{P}) \leq U(g, \mathcal{P})$.
    \item $L(-f, \mathcal{P}) = -U(f, \mathcal{P})$.
    \item For fixed $\lambda > 0$, $L(\lambda f, \mathcal{P}) = \lambda L(f, \mathcal{P})$ and $U(\lambda f, \mathcal{P}) = \lambda U(f, \mathcal{P})$.
    \item $U(f + g, \mathcal{P}) \leq U(f, \mathcal{P}) + U(g, \mathcal{P})$ and $L(f + g, \mathcal{P}) \geq L(f, \mathcal{P}) + L(g, \mathcal{P})$.
    \item $U(|f|, \mathcal{P}) - L(|f|, \mathcal{P}) \leq U(f, \mathcal{P}) - L(f, \mathcal{P})$.
    \item $U(f^2, \mathcal{P}) - L(f^2, \mathcal{P}) \leq 2[\sup_{x \in [a, b]} f(x)][U(f, \mathcal{P}) - L(f, \mathcal{P})]$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  All follow directly from the corresponding properties in \cref{infSupProperties} above.
\end{proof}
\subsection{Properties of Integrals}
\begin{lemma}
  Let $f, g: [a, b] \to \R$ be integrable functions. \label{integralProperties}
  We then have the following:
  \begin{enumerate}
    \item If $f(x) \leq g(x)\ \forall x \in [a, b]$, then $\int_{a}^{b} f(x) \d{x} \leq \int_{a}^{b} g(x) \d{x}$.
    \item For $\lambda \in \R$, $\lambda f(x)$ is integrable and $\int_{a}^{b} \lambda f(x) \d{x} = \lambda \int_{a}^{b} f(x) \d{x}$.
    \item $f + g$ is integrable and $\int_{a}^{b} (f(x) + g(x)) \d{x} = \int_{a}^{b} f(x) \d{x} + \int_{a}^{b} g(x) \d{x}$.
    \item $|f|$ is integrable and $\int_{a}^{b} |f| \d{x} \geq \abs{\int_{a}^{b} f(x) \d{x}}$.
    \item $fg$ is integrable.
  \end{enumerate}
\end{lemma}
\begin{remark}
  Property \textbf{iv} above is like a triangle inequality for integrals.
\end{remark}
\begin{proof}
  \begin{enumerate}
    \item From \cref{upperLowerProperties} \textbf{(i)} above, we have $L(f, \mathcal{P}) \leq L(g, \mathcal{P})$ for any partition $\mathcal{P}$ of $[a, b]$.
      Taking supremum of both sides over all $\mathcal{P}$, we have $I_{*}(f) \leq I_{*}(g)$.
      Since both are integrable $\int_{a}^{b} f(x) \d{x} = I_{*}(f) \leq I_{*}(g) = \int_{a}^{b} g(x) \d{x}$.
    \item If $\lambda = 0$ then the result follows trivially so assume $\lambda \neq 0$.
      For any $\varepsilon > 0$ since $f$ is integrable, $\exists \mathcal{P}$ such that $U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon$.
      We then have:
      \begin{align*}
        \int_{a}^{b} f(x) \d{x} \geq L(f, \mathcal{P}) \implies U(f, \mathcal{P}) < \int_{a}^{b} f(x) \d{x} + \varepsilon \\
        \int_{a}^{b} f(x) \d{x} \leq U(f, \mathcal{P}) \implies L(f, \mathcal{P}) > \int_{a}^{b} f(x) \d{x} - \varepsilon
      \end{align*}
      From \cref{upperLowerProperties}, if $\lambda > 0$, $L(\lambda f, \mathcal{P}) = \lambda L(f, \mathcal{P})$ and $U(\lambda f, \mathcal{P}) = \lambda U(f, \mathcal{P})$.
      We can then construct the following chain of inequalities:
      \[
        \lambda \int_{a}^{b} f \d{x} -\lambda \varepsilon \leq L(\lambda f, \mathcal{P}) \leq I_{*}(\lambda f) \leq I^{*}(\lambda f) \leq U(\lambda f, \mathcal{P}) \leq \lambda \int_{a}^{b} f \d{x} + \lambda \varepsilon
      \]
      But since $\varepsilon$ is arbitrary, we conclude that $I_{*}(\lambda f) = I^{*}(\lambda f) = \lambda \int_{a}^{b} f \d{x}$.
      So $\lambda f$ is integrable with $\int_{a}^{b} \lambda f \d{x} = \lambda \int_{a}^{b} f \d{x}$.

      Now for $\lambda = -1$, by \cref{upperLowerProperties} \textbf{(ii)}, we have:
      \begin{align*}
        L(-f, \mathcal{P}) = - U(f, \mathcal{P}) > -\int_{a}^{b} f(x) \d{x} - \varepsilon \\
        U(-f, \mathcal{P}) = - L(f, \mathcal{P}) < -\int_{a}^{b} f(x) \d{x} + \varepsilon
      \end{align*}
      So we can construct the following chain of inequalities:
      \[
        - \int_{a}^{b} f \d{x} - \varepsilon \leq L(-f, \mathcal{P}) \leq I_{*}(-f) \leq I^{*}(-f) \leq U(-f, \mathcal{P}) \leq - \int_{a}^{b} f \d{x} + \varepsilon
      \]
      Again, since $\varepsilon$ is arbitrary, we conclude that $I_{*}(-f) = I^{*}(-f) = - \int_{a}^{b} f \d{x}$.
      So $-f$ is also integrable with $\int_{a}^{b} -f \d{x} = -\int_{a}^{b} f \d{x}$.

      Since we have shown the result for $\lambda \geq 0$ and $\lambda = -1$, the result follows for all $\lambda \in \R$ by combining the above.
    \item By the same argument as in \textbf{(ii)}, since $f$ and $g$ are integrable $\exists \mathcal{P}_1, \mathcal{P}_2$ such that:
      \begin{align*}
        &\int_{a}^{b} f(x) \d{x} -\varepsilon \leq L(f, \mathcal{P}_1) \leq U(f, \mathcal{P}_1) \leq \int_{a}^{b} f(x) \d{x} + \varepsilon \\
        &\int_{a}^{b} g(x) \d{x} -\varepsilon \leq L(g, \mathcal{P}_2) \leq U(g, \mathcal{P}_2) \leq \int_{a}^{b} g(x) \d{x} + \varepsilon
      \end{align*}
      The partition $\mathcal{P} = \mathcal{P}_1 \cup \mathcal{P}_2$ is a refinement of $\mathcal{P}_1$ and $\mathcal{P}_2$ so using \cref{refinementInequality} and \cref{upperLowerProperties} \textbf{(iv)}, we have:
      \begin{align*}
        \int_{a}^{b} f(x) \d{x} + \int_{a}^{b} g(x) \d{x} - 2\varepsilon &\leq L(f, \mathcal{P}_1) + L(g, \mathcal{P}_2) \\
                                                                         &\leq L(f, \mathcal{P}) + L(g, \mathcal{P}) \\
                                                                         &\leq L(f + g, \mathcal{P}) \\
                                                                         &\leq I_{*}(f + g)
      \end{align*}
      By a similar argument $\int_{a}^{b} f \d{x} + \int_{a}^{b} g \d{x} + 2\varepsilon \geq I^{*}(f + g)$.
      Therefore, since $I_{*}(f + g) \leq I^{*}(f + g)$ from \cref{upperLowerIntegralRelation}, we have:
      \[
        \int_{a}^{b} f \d{x} + \int_{a}^{b} g \d{x} - 2\varepsilon \leq I_{*}(f + g) \leq I^{*}(f + g) \leq \int_{a}^{b} f \d{x} + \int_{a}^{b} g \d{x} + 2\varepsilon
      \]
      Thus as $\varepsilon$ can be arbitrarily small, we have $I_{*}(f + g) = I^{*}(f + g) = \int_{a}^{b} f \d{x} + \int_{a}^{b} g \d{x}$, as required.
    \item Using \cref{upperLowerProperties} \textbf{(v)}, for any partition $\mathcal{P}$ of $[a, b]$:
      \[
        U(|f|, \mathcal{P}) - L(|f|, \mathcal{P}) \leq U(f, \mathcal{P}) - L(f, \mathcal{P})
      \]
      Hence, if $f$ is integrable, so is $|f|$.
      Since $-|f(x)| \leq f(x) \leq |f(x)|\ \forall x \in [a, b]$, hence by \textbf{(i)}, we have:
      \begin{align*}
        - \int_{a}^{b} |f(x)| \d{x} \leq \int_{a}^{b} f(x) \d{x} \leq \int_{a}^{b} |f(x)| \d{x} \implies \abs{\int_{a}^{b} |f(x)| \d{x}} \geq \abs{\int_{a}^{b} f(x) \d{x}}
      \end{align*}
      $|f| \geq 0\ \forall x \implies \int_{a}^{b} |f| \d{x} \geq 0 \implies |\int_{a}^{b} |f| \d{x}| = \int_{a}^{b} |f| \d{x}$ so the result follows.
    \item We can rewrite $fg$ as $fg = \frac{1}{4}\left[(f + g)^2 - (f - g)^2\right]$.
      By \textbf{(ii)} and \textbf{(iii)} both $f + g$ and $f - g$ are integrable, so we just need to show that if some function $f$ is integrable, so is $f^2$.

      By \cref{upperLowerProperties} \textbf{(vi)}, for any partition $\mathcal{P}$ of $[a, b]$, we have:
      \[
        U(f^2 , \mathcal{P}) - L(f^2, \mathcal{P}) \leq 2 \sup_{x \in I} |f(x)| \cdot [U(f, \mathcal{P}) - L(f, \mathcal{P})]
      \]
      Since $f$ is integrable, we can make $U(f, \mathcal{P}) - L(f, \mathcal{P})$ arbitrarily small.
      Therefore, since $2 \sup_{I} f$ is just a non-negative constant, we can make $U(f^2, \mathcal{P}) - L(f^2, \mathcal{P})$ arbitrarily small and so $f^2$ from which the result follows as established.
  \end{enumerate}
\end{proof}
\begin{remark}
  From the proof of property \textbf{(ii)} above, we have the useful property that if $f: [a, b] \to \R$ is integrable, then $\forall \varepsilon > 0\ \exists \mathcal{P}$ of $[a, b]$ such that:
  \[
    \int_{a}^{b} f(x) \d{x} -\varepsilon \leq L(f, \mathcal{P}) \leq U(f, \mathcal{P}) \leq \int_{a}^{b} f(x) \d{x} + \varepsilon
  \]
\end{remark}
\section{Integration and Differentiation}
You have probably seen previously that differentiation is the inverse process of integration.
But why is this?

For this it is useful to represent the integral of $f: [a, b] \to \R$ from $a$ to $x$ as a function $F(x)$:
\[
  F: [a, b] \to \R,\ x \mapsto \int_{a}^{x} f(t) \d{t}
\]
We will first check that $F$ is continuous otherwise there is no chance it can be differentiable (see \cref{diffCont}).
\begin{proposition}[Integration is a Continuous Operation]
  If $f: [a, b] \to \R$ be Riemann integrable, then $F$ as defined above is continuous in $[a, b]$.
\end{proposition}
\begin{proof}
  \begin{align*}
    |F(x + h) - F(x)| &= \abs{\int_{a}^{x + h} f(t) \d{t} - \int_{a}^{x} f(t) \d{t}} \\
                      &= \abs{\int_{x}^{x + h} f(t) \d{t}} \text{ by \cref{integralAdditivity}}\\
                      &\leq \int_{x}^{x+h} |f(t)| \d{t} \\
                      &\leq \sup_{x \in [a, b]} |f| \int_{x}^{x+h} 1 \d{t} \\
                      &= h \sup_{x \in [a, b]} f
  \end{align*}
  Since $f$ is bounded, $\sup_{x \in [a, b]} f$ is a finite constant and so as $h \to 0$, $|F(x + h) - F(x)| \to 0$.
  Thus, $F$ is continuous on $[a, b]$.
\end{proof}
\begin{theorem}[Fundemental Theorem of Calculus -- Part 1]
  If $f: [a, b] \to \R$ is Riemann integrable and continuous at $x_0$ then $F$ as above is differentiable at $x_0$ and:
  \[
    F'(x_0) = \at{\deriv{}{x}\left[\int_{a}^{x} f(t) \d{t}\right]}{x = x_0}  = f(x_0)
  \]
\end{theorem}
\begin{proof}
  To do this, we can use the definition of differentiability involving an error term from \cref{derivativeAlternative}.
  Define $\varepsilon(h)$ as:
  \[
    \varepsilon(h) = \frac{F(x_0 + h) - F(x_0) - hf(x_0)}{|h|}
  \]
  We then want to show that $\varepsilon(h) \to 0$ as $h \to 0$.
  We can estimate the numerator as follows:
  \begin{align*}
    |F(x_0 + h) - F(x_0) - hf(x_0)| &= \abs{\int_{x_0}^{x + h} f(t) \d{t} - h f(x_0)} \\
                                    &= \abs{\int_{x_0}^{x_0+h} f(t) \d{x} - f(x_0)\int_{x_0}^{x_0+h} \d{t}} \\
                                    &= \abs{\int_{x_0}^{x_0+h} f(t) \d{x} - \int_{x_0}^{x_0+h} f(x_0) \d{t}} \text{ by \cref{integralProperties} \textbf{(ii)}}\\
                                    &= \abs{\int_{x_0}^{x_0+h} f(t) - f(x_0) \d{t}} \text{ by \cref{integralProperties} \textbf{(iii)}} \\
                                    &\leq \int_{x_0}^{x_0+h} |f(t) - f(x_0)| \d{t} \text{ by \cref{integralProperties} \textbf{(iv)}}
  \end{align*}
  Since $|f(t) - f(x_0)| \leq \sup_{t \in [x_0, x_0 + h]} |f(t) - f(x_0)|\ \forall t \in [x_0, x_0 + h]$, by \cref{integralProperties} \textbf{(i)} we have:
  \begin{align*}
    |F(x_0 + h) - F(x_0) - hf(x_0)| &\leq \sup_{t \in [0, h]}|f(x_0 + t) - f(x)| \cdot \int_{x_0}^{x_0 + h} 1 \d{t} \\
                                    &= h \cdot \sup_{t \in [0, h]} |f(x_0 + t) - f(x)|
  \end{align*}
  Thus $|\varepsilon(h)| \leq \sup_{t \in [0, h]} |f(x_0 + t) - f(x_0)| \to 0$ as $h \to 0$.

  Finally, by what we saw in \cref{derivativeAlternative}, the coefficient of $h$ in the numerator is the derivative at $x_0$ and so $F'(x_0) = f(x_0)$.
\end{proof}
\begin{example}
  Take $[a, b] = [-1, 1]$ and define:
  \[
    f(x) = \begin{cases}
    1 & \text{ if } x \geq 0\\
    -1 & \text{ if } x < 0
    \end{cases}
  \]
  This has a discontinuity at $x = 0$ but is piecewise continuous and so by \cref{piecewiseIntegrable} is Riemann integrable with:
  \[
    F(x) = \int_{-1}^{x} f(t) \d{t} = \begin{cases}
    -1-x & \text{ if } x\leq 0 \\
    x - 1 & \text{ if } x \geq 0
    \end{cases} = |x| - 1
  \]
  This is not differentiable at $x = 0$, which is where the discontinuity of $f$ was so the continuity of $f$ is necessary in the above.
\end{example}
\begin{corollary}
  If $f = g'$ is continuous on $[a, b]$, then $F(x) = \int_{a}^{x} f(t) \d{t} = g(x) - g(a)\ \forall x \in [a, b]$.
\end{corollary}
\begin{proof}
  $f$ is continuous so by the above $F$ is differentiable.
  \[
    \deriv{}{x}(F(x) - g) = f(x) - g'(x) = 0\ \forall x \in [a, b]
  \]
  So by \cref{derivativeIncreasingRelation}, we must have that $F - g$ is constant on $[a, b]$, i.e.
  \[
    F(x) - g(x) = F(a) - g(a) = -g(a)\ \forall x \in [a, b] \implies F(x) = g(x) - g(a)\ \forall x \in [a, b]
  \]
\end{proof}
We can prove a stronger (i.e. weaker assumptions) and more general result of the FTC.
\begin{theorem}[Fundemental Theorem of Calculus -- Part 2]
  If $f: [a, b] \to \R$ is Riemann integrable and there exists a differentiable $F: [a, b] \to \R$ such that $F'(x) = f(x)$, then:
  \[
    \int_{a}^{b} f(t) \d{t} = \int_{a}^{b} F'(t) \d{t} = F(b) - F(a)
  \]
\end{theorem}
\begin{proof}
  By assumption $f$ is integrable so $\exists \mathcal{P}$ partition of $[a, b]$ such that $U(f, \mathcal{P}) - L(f, \mathcal{P}) < \varepsilon$.
  We can also apply the mean value theorem \cref{MVC} to show that for a pair of adjacent points $x_{j - 1}$ and $x_{j}$ in the partition, we have
  \[
    f(t_j) = F'(t_j) = \frac{F(x_j) - F(x_{j - 1})}{x_j - x_{j - 1}} \implies F(x_j) - F(x_{j - 1}) = f(t_j) (x_j - x_{j - 1})
  \]
  for some $t_j \in [x_{j - 1}, x_j]$.
  We can repeat this for each pair of adjacent points in the partition.

  If we take sums from 1 to $n$ of both sides, the left hand side telescopes and so we have:
  \[
    F(b) - F(a) = \sum_{j = 1}^{n} f(t_j)(x_j - x_{j - 1})
  \]
  For each $j$, $t_j \in [x_{j - 1}, x_j] \implies \inf_{[x_j, x_{j - 1}]} f \leq f(t_j) \leq \sup_{[x_j, x_{j - 1}]} f$.
  Therefore the right hand side of this sum is between $L(f, \mathcal{P})$ and $U(f, \mathcal{P})$, thus $L(f, \mathcal{P}) \leq F(b) - F(a) \leq U(f, \mathcal{P})$.
  Applying \cref{infSupProperties} \textbf{(i)}, we then have:
  \[
    I_{*}(f) = \sup_{\mathcal{P}} L(f, \mathcal{P}) \leq F(b) - F(a) \leq \inf_{\mathcal{P}} U(f, \mathcal{P}) = I^{*}(f)
  \]
  Since $f$ is integrable, $I_{*}(f) = I^{*}(f) = \int_{a}^{b} f(x) \d{x}$, by the above inequality $\int_{a}^{b} f(x) \d{x} = F(b) - F(a)$, as required.
\end{proof}
