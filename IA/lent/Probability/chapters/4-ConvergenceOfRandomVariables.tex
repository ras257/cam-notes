\chapter{Convergence of Random Variables}
\section{Convergence in Probability}
\begin{definition}[Convergence in Probability]
  A sequence of random variables $(X_n: n \in \N)$ \textit{converges to a random variable $X$ in probability} if:
  \[
    \forall \varepsilon > 0,\ \P(|X_n - X| > \varepsilon) \to 0 \text{ as } n \to \infty
  \]
  we then write:
  \[
    X_n \xrightarrow{P} X \text{ or } \underset{n\to\infty}{\operatorname{plim}} X_n = X
  \]
\end{definition}
\begin{theorem}[Weak Law of Large Numbers]
  Let $(X_n)$ be independent and identically distributed random variables with finite $\mu = \E[X_1]$.

  If we set $S_n = X_1 + \cdots + X_n$ for $n \in \N$, then:
  \[
    \frac{S_n}{n} \xrightarrow{P} \mu \text{ as } n \to \infty
  \]
  That is:
  \[
    \forall \varepsilon > 0,\ \P\left(\abs{\frac{S_n}{n} - \mu} > \varepsilon\right) \to 0 \text{ as } n \to \infty
  \]
\end{theorem}
\begin{remark}
  In the definition of convergence in probability, we specified that a sequence converges to \textbf{random variable} $X$, here $\mu$ is just a constant random variable so we can still apply the definition.
\end{remark}
\begin{proof}
  For ease, in this proof, we will assume $\sigma^2 = \Var(X_1) < \infty$.
  \begin{align*}
    \P\left(\abs{\frac{S_n}{n} - \mu} > \varepsilon\right) &= \P(|S_n - n\mu| > n\varepsilon) \\
                                                           &\leq \frac{\E[|S_n - n\mu|^2]}{n^2 \varepsilon^2} \text{ using Markov's -- \cref{markovs}} \\
                                                           &= \frac{\Var(S_n)}{n^2 \varepsilon^2} \text{ as $\E[S_n] = n\mu$}
  \end{align*}
  Since the $X_i$ are i.i.d, we have that $\Var(S_n) = n\sigma^2$ so:
  \[
    \P\left(\abs{\frac{S_n}{n} - \mu} > \varepsilon\right) \leq \frac{n \sigma^2}{n^2 \varepsilon^2} = \frac{\sigma^2}{n \varepsilon^2} \to 0 \text{ as } n \to \infty
  \]
  as required.
\end{proof}
\section{Convergence Almost Surely}
\begin{definition}
  Let $(X_n)_{n \in \N}$ be a sequence of random variables at let $X$ be another random variable.
  We say that \textit{$X_n$ converges to $X$ almost surely} or \textit{with probability 1} if:
  \[
    \P\left(\lim_{n \to 0} X_n = X\right) = 1
  \]
  We then write $X_n \to X$ as $n \to \infty$ \textit{almost surely}.
\end{definition}
\begin{remark}
  When we say:
  \[
    \P\left(\lim_{n \to \infty} X_n = X\right) = 1
  \]
  We mean that:
  \[
    \exists A \in \salg \text{ s.t. } \P(A) = 1 \text{ and } \forall \omega \in A\ \forall \varepsilon > 0 \ \exists n_0(\varepsilon, \omega) \text{ s.t. } \forall n\geq n_0\ |X_n(\omega) - X(\omega)| \leq \varepsilon
  \]
  So there is a set $A$ with probability one where the numerical sequence $(X_n(\omega))_n$ converges to $X(\omega)$ for all $\omega \in A$.
\end{remark}
\begin{proposition}[Convergence almost surely $\implies$ Convergence in probability]
  If $X_n \to 0$ as $n \to \infty$ almost surely, then $X_n \xrightarrow{P} 0$ as $n \to \infty$.
\end{proposition}
\begin{proof}
  We need to show that $\forall \varepsilon > 0\ \P(|X_n| > \varepsilon) \to 0$ as $n \to \infty$, or equivalently, $\forall \varepsilon > 0\ \P(|X_n| \leq \varepsilon) \to 1$ as $n \to \infty$.

  To show that $\P(|X_n| \leq \varepsilon) \to 1$, we want to bound this below by something that increases to 1 in the limit.
  We can start by bounding it as:
  \[
    \P(|X_n| \leq \varepsilon) \geq \P\left(\bigcap_{m = n}^{\infty} \{|X_m| \leq \varepsilon\}\right)
  \]
  since $\bigcap_{m = n}^{\infty} \{|X_m| \leq \varepsilon \} \subseteq \{|X_n| \leq \varepsilon \}$.
  Then let $A_n = \bigcap_{m = n}^{\infty} A_n$.
  This is an increasing sequence of events as $A_{n} \subseteq A_{n + 1}\ \forall n$ so by continuity of probability measures (\cref{continuityProbMeasures}):
  \begin{align*}
    \P(A_n) \to \P\left(\bigcup_{n = 0}^{\infty} A_n\right) &= \P\left(\bigcup_{n = 0}^{\infty} \bigcap_{m = n}^{\infty} \{|X_n| \leq \varepsilon\}\right) \\
                                                            &= \P(|X_m| \leq \varepsilon \text{ for all $m$ sufficiently large}) \\
                                                            &\geq \P(\forall \varepsilon > 0\ |X_m| \leq \varepsilon \text{ for all $m$ sufficiently large}) \\
                                                            &= 1 \text{ as $X_n \to 0$ almost surely}
  \end{align*}
  Thus $\P(|X_n| \leq \varepsilon) \to 1$ as $n \to \infty$ so $X_n \xrightarrow{P} 0$ as $n \to \infty$.
\end{proof}
\begin{theorem}[Strong Law of Large Numbers -- SLLN]
  Let $(X_n)$ be independent and identically distributed random variables with finite $\mu = \E[X_1]$.
  If we set $S_n = X_1 + \cdots + X_n$ for $n \in \N$, then: \label{SLLN}
  \[
    \frac{S_n}{n} \to \mu \text{ as } n \to \infty \text{ almost surely}
  \]
\end{theorem}
\begin{proof}
  \nonexaminable
  For ease, in this proof, we will assume that $\E[X^4_1] < \infty$.

  Set $Y_i = X_i - \mu$ so that $\E[Y_i] = 0$.
  We can then check that $\E[Y^4_i]$ is also finite:
  \[
    \E[Y^4_i] = \E[(X_1 - \mu)^4] \leq 2^{4} \cdot (\E[X^4_1] + \mu^4) < \infty
  \]
  Define $S_n = \sum_{i = 1}^{n} Y_i$, we need to show that $\frac{S_n}{n} \to 0$  as $n \to \infty$ almost surely.
  To do this, our goal is to show that the series $\sum_{n = 1}^{\infty} (\frac{S_n}{n})^4$ is finite with probability 1.
  By the $n$-th term test (See Analysis I), if a series converges, its terms must go to 0 and therefore if $\sum_{n = 1}^{\infty} (\frac{S_n}{n})^4$ is finite with probability 1, $\frac{S_n}{n} \to 0$ as $n \to \infty$ with probability 1.

  To show the sum is finite with probability 1, it suffices to show that its expectation is finite since this implies that it is finite with probability one.

  Since $(\frac{S_n}{n})^4$ are non-negative, by \cref{sumExpectationSwap}, we can use the linearity of expectation on the infinite sum:
  \begin{align*}
    \E\left[\sum_{n = 1}^{\infty} \left(\frac{S_n}{n}\right)^4\right] &= \sum_{n = 1}^{\infty}\frac{1}{n^4} \E[S^4_n]
  \end{align*}
  We therefore want to bound $\E[S^4_n]$:
  \begin{align*}
    \E[S^4_n] &= \E[(Y_1 + \cdots Y_n)^4] \\
              &= \E\left[\sum_{i = 1}^{n} Y^4_i + \binom{4}{2}\sum_{1 \leq i < j \leq n}^{n} Y^2_i Y^2_j + R\right]
  \end{align*}
  where $R$ is a sum of terms of the form $Y^3_iY_j$, $Y^2_iY_jY_k$ and $Y_iY_jY_kY_l$ with distinct indices $i, j, k, l$.
  Since the $Y_i$ are independent and $\E[Y_i] = 0$, we then have $\E[R] = 0$.
  Note also that the $\binom{4}{2}$ originates from the fact that, for each $i < j$, we pick 2 of the 4 brackets to pick the $Y_i$ terms from, and then pick the $Y_j$ terms from the other two.

  For the terms with two squares:
  \begin{align*}
    \E[Y^2_iY^2_j] &= (\E[Y^2_1])^2 \text{ as they are i.i.d}\\
                   &\leq \E[Y^4_1] < \infty \text{ using Jensen's inequality -- \cref{jensensInequality}}
  \end{align*}
  Therefore we can bound $\E[S^4_n]$ as:
  \begin{align*}
    \E[S^4_n] &\leq n \E[Y^4_1] + 6 \cdot \binom{n}{2} \cdot \E[Y^4_1] \text{ as there are $\binom{n}{2}$ ways to pick $i < j$} \\
              &\leq 3n^2 \E[Y^4_1]
  \end{align*}
  So:
  \[
    \E\left[\sum_{n = 1}^{\infty} \left(\frac{S_n}{n}\right)^4\right] = \sum_{n = 1}^{\infty} \frac{\E[S^4_n]}{n^4} \leq \sum_{n = 1}^{\infty} \frac{3}{n^2} \cdot \E[Y^4_1] < \infty
  \]
  as $\sum_{n = 1}^{\infty} \frac{1}{n^2}$ converges ans $\E[Y^4_1] < \infty$.
\end{proof}
\section{Central Limit Theorem}
\subsection{Prelude}
Let $X_1, \ldots, X_n$ be independent and identically distributed random variables with $\mu = \E[X_1] < \infty$ and $\sigma^2 = \Var(X_1) < \infty$.
Again, define $S_n = X_1 + \cdots X_n$.
By the strong law of large numbers above, we expect $S_n \approx n\mu$ for large $n$.

But how does $S_n$ fluctuate around $n \mu$?
We can calculate the variance and expectation of these fluctuations:
\[
  \Var(S_n - n\mu) = \Var(S_n) = n \sigma^2 \text{ and } \E[S_n - n\mu] = 0
\]
The random variable $\frac{S_n - n\mu}{\sqrt{n \sigma^2}}$ then has expectation 0 and variance 1.
We can rewrite this random variable as:
\[
  \frac{S_n/n - \mu}{\sqrt{\sigma^2/n}} = \frac{S_n/n - \mu}{\sqrt{\Var(S_n/n - \mu)}}
\]
since $\Var(S_n/n - \mu) = \sigma^2/n$.
So if we take $\frac{S_n}{n} - \mu$, which we know converges to 0 almost surely, and then divide it by the square-root of its variance, we get a random variable with expectation 0 and variance 1.

Consider the case where the $X_i$ are independent and identically distributed  as $\normal(\mu, \sigma^2)$.
Since the sum of Gaussians is also a Gaussian (see \cref{sumOfNormals}), we have $S_n - n\mu \sim \normal(0, n\sigma^2)$ and therefore $\frac{S_n - n\mu}{\sigma \sqrt{n}} \sim \normal(0, 1)$.
So in this case, it follows a standard normal.
Although we do not yet know what the distribution of this is when working with a general distribution for $X_i$, this has the same expectation and variance as when we were just using any distribution for the $X_i$.

As we will see, the normal distribution is \textit{universal} as it appears as the limit of the distribution $\frac{S_n - n\mu}{\sigma \sqrt{n}}$, regardless of the distribution of $X_i$.
This is concept of the \textit{central limit theorem}.
\subsection{Convergence in Distribution}
\begin{definition}[Convergence in Distribution]
  A sequence of random variables $(X_n)_{n \in N}$ \textit{converges to a random variable $X$ in distribution} as $n \to \infty$ if:
  \[
    F_{X_n}(x) \to F_X(x) \text{ as $n \to \infty$}\ \forall x \in \R \text{ that are continuity points of $F_X$}
  \]
  where $F_{X_n}(x) = \P(X_n \leq x)$ is the probability distribution function.
  We then write $X_n \xrightarrow{(d)} X$ as $n \to \infty$.
\end{definition}
\begin{remark}
  In \cref{pdfProperties}, we saw that left limits always exist for $F$ and $F$ is always right continuous.
  This means that the continuity points are just the points where the left limit equals the right limit (which, by continuity, is the value of the function at that point).
\end{remark}
\begin{remark}[Limit Inferior and Limit Superior]
  For following proof, it is useful to introduce the following definitions for a sequence $(x_n)$ on $\R$:
  \begin{align*}
    \liminf_{n \to \infty} x_n := \lim_{n \to \infty} \left(\inf_{m \geq n} x_m\right) \\
    \limsup_{n \to \infty} x_n := \lim_{n \to \infty} \left(\sup_{m \geq n} x_m\right)
  \end{align*}
  These are useful as they always exist for a sequence on $\R$.

  Notably, a sequence $(X_n)$ in $[-\infty, \infty]$ converges if and only if its limit superior and limit inferior are equal.
\end{remark}
\begin{proposition}[Convergence in probability $\implies$ Convergence in distribution]
  If $X_n \xrightarrow{P} X$  as $n \to \infty$, then $X_n \xrightarrow{(d)} X$ as $n \to \infty$.
\end{proposition}
\begin{proof}
  Let $x$ be an arbitrary continuity point of $F_X$.
  We need to show that $\lim_{n \to \infty} F_{X_n}(x) = F_X(x)$.

  Let $\varepsilon > 0$ be arbitrary and consider the event $\{X \leq x - \varepsilon\}$.
  Either $X_n \leq x$ or, if $X_n > x$, since we also have $X \leq x - \varepsilon$, we must then have $|X_n - X| > \varepsilon$.
  So we then have:
  \[
    \{X \leq x - \varepsilon\} \subseteq \{X_n \leq x\} \cup \{|X_n - X| > \varepsilon\}
  \]
  Similarly for the event $\{X_n \leq x\}$, either $X \leq x + \varepsilon$, or if $X < x + \varepsilon$, since we also have $X_n \leq x$, we must have $|X_n - X| > \varepsilon$.
  So we then have:
  \[
    \{X_n \leq x\} \subseteq \{X \leq x + \varepsilon\} \cup \{|X_n - X| > \varepsilon\}
  \]
  Therefore by probability of
  \begin{align*}
    \P(X \leq x - \varepsilon) &\leq \P(\{X_n \leq x\} \cup \{|X_n - x| > \varepsilon\}) \text{ by \cref{propertiesOfProbMeasure}} \\
                               &\leq \P(X_n \leq x) + \P(|X_n - x| > \varepsilon) \text{ by \cref{countableSub}} \\
    \implies F_X(x - \varepsilon) &\leq \P(X_n \leq x) + \P(|X_n - x| > \varepsilon) \\
                                  &\leq \liminf_{n \to \infty} [\P(X_N \leq x) + \P(|X_n - X| > \varepsilon)] \\
                                  &= \liminf_{n \to \infty} \P(X_n \leq x) \text{ since $X_n \xrightarrow{P} X$} \\
                                  &= \liminf_{n \to \infty} F_{X_n}(x)
  \end{align*}
  By a similar chain of inequalities:
  \begin{align*}
    F_{X_n}(x) &= \P(X_n \leq x) \leq \P(X \leq x + \varepsilon) + \P(|X_n - X| > \varepsilon) \\
    \implies \limsup_{n \to \infty} F_{X_n(x)} &\leq \limsup_{n \to \infty} [(F_X(x + \varepsilon)) + \P(|X_n - X| > \varepsilon) \\
                                               &= F_X(X + \varepsilon) \text{ since $X_n \xrightarrow{P} X$} \\
  \end{align*}
  Combining the two inequalities, we then have:
  \[
    F_X(x - \varepsilon) \leq \liminf_{n \to \infty} F_{X_n}(x) \leq \limsup_{n \to \infty} F_{X_n}(x) \leq F_{X}(x + \varepsilon)
  \]
  Finally, taking $\varepsilon \to 0$, since $x$ is a continuity point of $F_{X}$, we have:
  \[
    F_{X}(x) \leq \liminf_{n \to \infty} F_{X_n}(x) \leq \limsup_{n \to \infty} F_{X_n}(x) \leq = F_{X}(x)
  \]
  Therefore:
  \[
    \liminf_{n \to \infty} F_{X_n}(x) = \limsup_{n \to \infty} F_{X_n}(x) = F_{X}(x) \implies \lim_{n \to \infty} F_{X_n}(x) = F_{X}(x)
  \]
  as required.
\end{proof}
\begin{theorem}[Continuity Property of mgfs]
  Let $(X_n)$ be random variables with moment generating functions $m_n(\theta) = \E[e^{\theta X_n}]$ for $\theta \in \R$ and let $X$ be a random variable with mgf $m(\theta) = \E[e^{\theta X}]$.
  Further, assume that that $m(\theta) < \infty$ for some $\theta \neq 0$. \label{mgfCty}

  We then have that if $m_n(\theta) \to m(\theta)$ as $n \to \infty\ \forall \theta \in \R$, then $X_n \xrightarrow{(d)} X$ as $n \to \infty$.
\end{theorem}
\begin{proof}
  Provided without proof.
\end{proof}
\subsection{Central Limit Theorem}
\begin{theorem}[Central Limit Theorem]
  If $(X_n)_{n \in \N}$ are independent and identically distributed random variables with $\mu = \E[X_1] < \infty$ and $\sigma^2 = \Var(X) < \infty$ and we define $S_n = X_1 + \cdots + X_n$, then: \label{CLT}
  \[
    \frac{S_n - n\mu}{\sigma \sqrt{n}} \xrightarrow{(d)} Z
  \]
  where $Z \sim \normal(0, 1)$.

  In other words, for all $x \in \R$:
  \[
    \P\left(\frac{S_n - n\mu}{\sigma \sqrt{n}} \leq x\right) \to \int_{-\infty}^{x} \frac{e^{-\frac{y^2}{2}}}{\sqrt{2\pi}} \d{y} = \Phi(x)
  \]
\end{theorem}
\begin{proof}\par
  \textbf{Shifting the Random Variables}\par
  To make the proof easier, we define $Y_i = \frac{X_i - \mu}{\sigma}$ so that $\E[Y_i] = 0$ and $\Var(Y_i) = 1$.
  Therefore, it is enough to prove that the theorem holds for a sequence of i.i.d random variables $(X_i)_{n \in \N}$ with mean 0 and variance 1.
  Thus, we need to show that $\frac{S_n}{\sqrt{n}} \xrightarrow{(d)} \normal(0, 1)$ as $n \to \infty$.

  \textbf{Using the mgf}\par
  For ease, in this proof, we  will assume that $\exists \delta > 0$ such that $\E[e^{\delta X_1}] + \E[e^{-\delta X_1}] < \infty$.
  By the continuity property of moment generating functions (\cref{mgfCty}), it suffices to show that
  \[
    \E[e^{\theta \frac{S_n}{\sqrt{n}}}] \xrightarrow{n\to \infty} \E[e^{\theta Z}] = \exp\left(\frac{\theta^2}{2}\right)\ \forall \theta \in \R
  \]
  using the mgf of $Z$ from \cref{mgfNormal}.

  Setting $m(\theta) = \E[e^{\theta X_1}]$, we can factor $\E[e^{\theta \frac{S_n}{\sqrt{n}}}]$ as:
  \begin{align*}
    \E[e^{\theta \frac{S_n}{\sqrt{n}}}] &= \E\left[e^{\frac{\theta}{\sqrt{n}}X_1 + \cdots \frac{\theta}{\sqrt{n}}X_n}\right] \\
                                        &= \E[e^{\frac{\theta}{\sqrt{n}}X_1}]^{n} \text{ as they are i.i.d} \\
                                        &= \left[m\left(\frac{\theta}{\sqrt{n}}\right)\right]^{n}
  \end{align*}
  So we want to show that $[m(\theta/\sqrt{n})]^{n} \to e^{\theta^2/2}$ as $n \to \infty\ \forall \theta \in \R$.

  We can then use the power series definition of the exponential:
  \begin{align*}
    m\left(\frac{\theta}{\sqrt{n}}\right) &= \E\left[e^{\frac{\theta}{\sqrt{n}}X_1}\right] \\
                                          &= \E\left[1 + \frac{\theta}{\sqrt{n}}X_1 + \frac{\theta^2}{n}X^2_1 \cdot \frac{1}{2} + \sum_{k = 3}^{\infty} \frac{\theta^{k}}{(\sqrt{n})^{k}} \cdot \frac{1}{k!}X^{k}_1\right] \\
                                          &= 1 + \frac{\theta^2}{2n} + \E\left[\sum_{k = 3}^{\infty} \frac{\theta^{k}}{(\sqrt{n})^{k}} \cdot \frac{1}{k!}X^{k}_1\right] \text{ since $\E[X_1] = 0$ and $\E[X^2_1] = 1$}
  \end{align*}
  We know that $(1 + \frac{\theta^2}{2n})^{n} \to e^{\theta^2/2}$ as $n \to \infty\ \forall \theta \in \R$, so we claim that:
  \[
    \abs{\E\left[\sum_{k =  3}^{\infty} \theta^{k} \cdot \frac{1}{k!}X^{k}_1\right]} = o(|\theta|^2) \text{ as } \theta \to 0
  \]
  If this claim is true, then we can conclude that:
  \[
    m\left(\frac{\theta}{\sqrt{n}}\right) = 1 + \frac{\theta^2}{2n} + o\left(\frac{\theta^2}{n}\right) \text{ as $n \to \infty$}
  \]
  and hence:
  \[
    \left[m\left(\frac{\theta}{\sqrt{n}}\right)\right]^{n} \to e^{\theta^2 / 2} \text{ as } n \to \infty\ \forall \theta \in \R
  \]
  \textbf{Proof of Order Claim}\\
  Let $\theta \in \R$ be arbitrary such that $|\theta| \leq \frac{\delta}{2}$.
  \begin{align*}
    \abs{\E\left[\sum_{k = 3}^{\infty} \frac{\theta^{k}X^{k}_1}{k!}\right]} &\leq \E\left[\sum_{k = 3}^{\infty} \frac{|\theta X_1|^{k}}{k!}\right] \text{ by \cref{jensensInequality}} \\
                                                                            &= \E\left[|\theta X_1|^3 \sum_{k = 0}^{\infty} \frac{|\theta X_1|^{k}}{(k + 3)!}\right] \\
                                                                            &\leq \E\left[|\theta X_1|^3 \sum_{k = 0}^{\infty} \frac{|\theta X_1|^{k}}{k!}\right] \\
                                                                            &= \E[|\theta X_1|^3 e^{|\theta X_1|}] \\
                                                                            &\leq \E[|\theta X_1|^3 \cdot e^{\frac{\delta}{2}|X_1|}] \text{ as $|\theta| \leq \frac{\delta}{2}$}
  \end{align*}
  We can rewrite $|\theta X_1|^3$ as:
  \[
    |\theta X_1|^3 = |\theta|^3 \cdot \frac{3!}{\left(\frac{\delta}{2}\right)^3} \cdot \frac{\abs{\frac{\delta}{2} X_1}^3}{3!} \leq C \cdot |\theta|^3 \cdot e^{\frac{\delta}{2}|X_1|}
  \]
  where $C = 3! \cdot 2^{3}/\delta^{3}$ since $e^{|y|} \geq \frac{|y|^{n}}{n!}\ \forall n \in \N$.

  Finally, we then have the bound:
  \begin{align*}
    \E[|\theta X_1|^3 \cdot e^{\frac{\delta}{2}|X_1|}] &\leq C \cdot |\theta|^3 \cdot \E[e^{\frac{\delta}{2}|X_1|} \cdot e^{\frac{\delta}{2}|X_1|}] \\
                                                       &= C \cdot |\theta|^3 \cdot \E[e^{\delta|X_1|}] \\
                                                       &\leq C \cdot |\theta|^3 \cdot \underbrace{\E[e^{\delta X_1} + e^{-\delta X_1}]}_{< \infty}
  \end{align*}
  Therefore:
  \[
    \abs{\E\left[\sum_{k = 3}^{\infty} \frac{\theta^{k}X^{k}_1}{k!}\right]} \leq C' \cdot |\theta|^3
  \]
  for some finite constant $C'$, and so this is $o(|\theta|^2)$ as $\theta \to 0$.
\end{proof}
\subsection{Applications}
The central limit theorem tells us that:
\[
  \frac{S_n - n\mu}{\sigma \sqrt{n}} \xrightarrow{(d)} \normal(0, 1)
\]
So for ``large'' $n$, we have:
\[
  S_n \approx \normal(n\mu, \sigma^2n) \text{ or equivalently } S_n \approx n\mu + \sigma\sqrt{n} Z
\]
where $Z \sim \normal(0, 1)$.
\subsubsection{Binomial Distribution}
Suppose $S_n \sim \binomial(n, p)$.
In \cref{binomialAsBernSum}, we saw that can express $S_n$ as a sum of $n$ $(X_i)$ which are independent and identically distributed Bernoulli random variables, $\operatorname{Ber}(p)$.

Using the central limit theorem, since $\Var(X_1) = p(1 - p)$ and $\E[X_1] = p$, we have:
\[
  \frac{S_n - np}{\sqrt{np(1 - p)}} \xrightarrow{(d)} \normal(0, 1) \text{ as } n \to \infty
\]
Thus, for large $n$, we can approximate a binomial distribution as $S_n \approx \normal(np, np(1 - p))$.

We also saw in \cref{binomialLimitPoisson}, that if $S_n \sim \binomial(n, \frac{\lambda}{n})$ for $\lambda > 0$ then:
\[
  \P(S_n \leq x) \xrightarrow{n \to \infty} \P(\poisson(\lambda) = x)
\]
Note that the central limit theorem does not apply here since the probability of the binomial distribution is a function of $n$ and so it cannot be expressed as a sum of i.i.d Bernoulli random variables.
\subsubsection{Poisson Distribution}
Suppose $S_n \sim \poisson(n)$.
In \cref{poissonConvolution}, we saw that if $X \sim \poisson(\lambda)$ and $Y \sim \poisson(\mu)$ are independent then $X + Y \sim \poisson(\lambda + \mu)$.
This means that we can express $S_n$ as a sum of $n$ $(X_i)$ i.i.d $\poisson(1)$ random variables.

Using the central limit theorem, since $\Var(X_i) = 1$ and $\E[X_i] = 1$:
\[
  \frac{S_n - n}{\sqrt{n}} \xrightarrow{(d)} \normal(0, 1) \text{ as } n \to \infty
\]
Thus, for large $n$, we can approximate a Poisson distribution as $S_n \approx \normal(n, n)$.
\subsection{Sampling Error}
Suppose we hold a referendum and a proportion $p$ of individuals is inclined to vote ``Yes''.
That is, every individual votes ``Yes'' with probability $p$ and ``No'' with probability $1 - p$.

It is natural to want to estimate $p$ but perhaps we do not want to ask the whole population.
We can instead take a sample of $N$ individuals at random and record their votes.
Let $X_i$ take the value 1 if the $i$-th individual voted ``Yes'' and 0 otherwise.
Finally, let $S_N$ be the total number of ``Yes'' votes, i.e. $S_N = X_1 + \cdots + X_N$.

Since $S_N$ is the sum of $N$ i.i.d Bernoulli random variables, as we saw in \cref{binomialAsBernSum}, $S_N \sim \binomial(n, p)$.
So using the strong law of large numbers (\cref{SLLN}), we have that:
\[
  \hat{p}_{N} = \frac{S_N}{N} \to p \text{ as $N \to \infty$}
\]
That is, as the size of the sample increases, $\hat{p}_{N}$ approaches $p$.

So that we do not need to sample the whole population, we want to determine $N$ such that our estimate $\hat{p}_N$ for $p$ is within some tolerance of $p$.
For example, suppose that we want to estimate $p$ to an accuracy of $\pm 4\%$ with probability greater than $0.99$.
We then want to find $N$ such that:
\[
  \P(|\hat{p}_{N} - p| \geq 0.04) \leq 0.01
\]
By the Central Limit Theorem (\cref{CLT}), for large $N$, $S_N \approx Np + \sqrt{Np(1 - p)Z}$ where $Z \sim \normal(0, 1)$.
We can then rewrite $\hat{p}_N$ using this:
\[
  \hat{p}_{N} = \frac{S_N}{N} \approx p + \frac{\sqrt{p(1 - p)}}{\sqrt{N}}Z
  \implies |\hat{P}_{N} - p| \approx \frac{\sqrt{p(1 - p)}}{\sqrt{N}} |Z|
\]
Furthermore, as we saw in \cref{normalSymmetryExample}, due to the symmetry of the density function for the normal distribution:
\[
  \P(|Z| \geq z) = 2(1 - \Phi(z))
\]
If we take $z = 2.58$, then either using a statistical table or numerical integration, we see that $\Phi(2.58) \approx 0.995$ and so $\P(|Z| \geq 2.58) = 0.01$.

Since we want to find $N$ such that:
\[
  \P\left(\frac{\sqrt{p(1 - p)}}{N} |Z| \geq 0.04\right) \leq 0.01
\]
we therefore need:
\[
  \frac{\sqrt{N}}{\sqrt{p(1 - p)}} \cdot 0.04 \geq 2.58 \implies N \geq 4160.25 \cdot p(1 - p)
\]
For $p \in [0, 1]$, $p(1 - p)$ attains a minimum value of $\frac{1}{4}$ and so $N \geq 1040$ is sufficiently large.
