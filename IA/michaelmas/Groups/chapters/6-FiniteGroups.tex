\chapter{Small Finite Groups}
We have already seen some nice theorems about finite groups:
\begin{itemize}
  \item Lagrange's Theorem
  \item Orbit-Stabiliser Theorem
  \item Cayley's Theorem
  \item Cauchy's Theorem
\end{itemize}
We will now spend some time thinking about small examples of finite groups.
\section{Groups of Order 1}
A group of order 1 has only one element, which must be the identity.
Any singleton set can be a group of order 1 with the binary operation $e \cdot e = e$.
All such groups are isomorphic, thus, up to isomorphism, there is only one group of order 1, the trivial group:
\[
  |G| = 1 \iff G \cong 1 = \{e\}
\]
\section{Groups of Prime Order}
From \cref{primeOrderGroup}, all groups of prime order must be cyclic.
Thus, if $|G| = p$, where $p$ is prime, then $G \cong C_p$.
For example:
\begin{align*}
  |G| = 2 &\iff G \cong C_2 \\
  |G| = 3 &\iff G \cong C_3 \\
  |G| = 5 &\iff G \cong C_5 \\
  |G| = 7 &\iff G \cong C_7
\end{align*}
and so on.
\section{Groups of Order 4}
If $|G| = 4$, we have $C_4$ as an example, but is there any other groups of order 4 up to isomorphism?
The direct product is useful for investigating this.
\begin{definition}[Direct Product]
  If $G$ and $H$ are groups, then we define their \textit{direct product} as:
  \[
    G \times H = \{(g, h) : g \in G, h \in H\}
  \]
  with group operation carried out component wise:
  \[
    (g_1, h_1) \cdot (g_2, h_2) = (g_1g_2, h_1h_2)
  \]
  This is a group as $(e_G, e_H)$ is the identity and $(g, h)^{-1} = (g^{-1}, h^{-1})$.
\end{definition}
\begin{example}[Klein 4-Group]
  The group:
  \[
    K_4 = C_2 \times C_2
  \]
  is often called the \textit{klein 4-group}.
  You can also think of it as $D_4$.

  Note that, for every $(a, b) \in K_4$,
  \[
    (a, b)^2 = (a^2, b^2) = e
  \]
  so every element has order at most 2.
  In particular, this means that $K_4 \centernot\cong C_4$.
\end{example}
\begin{theorem}[Direct Product Theorem]
  If $H_1, H_2 \leq G$ and: \label{directProductTheorem}
  \begin{enumerate}
    \item $H_1 \cap H_2 = \{e\}$
    \item $h_1 h_2 = h_2 h_1$ for all $h_1 \in H_1, h_2 \in H_2$
    \item $G = H_1 H_2$, that is, for all $g \in G$ there is $h_1 \in H_1, h_2 \in H_2$ such that $g = h_1 h_2$
  \end{enumerate}
  then $G\cong H_1 \times H_2$.
\end{theorem}
\begin{proof}
  Define the map:
  \[
    \Phi: H_1 \times H_2 \to G, (h_1, h_2) \mapsto h_1h_2
  \]
  We will start by showing $\Phi$ is a homomorphism.
  For any $h_1, h_1' \in H, h_2, h_2' \in H_2$:
  \begin{align*}
    \Phi(h_1, h_2)\Phi(h_1', h_2') &= h_1 h_2 h_1' h_2' \\
                                   &= h_1 h_1' h_2 h_2' \text{ (by \textbf{ii})}\\
                                   &= \Phi(h_1h_1', h_2h_2') \\
                                   &= \Phi((h_1, h_2)(h_1', h_2'))
  \end{align*}
  so $\Phi$ is a homomorphism.

  Surjectivity of $\Phi$ is immediate from \textbf{iii}.
  We now just need to show $\Phi$ is injective.
  Recall that we can do this by showing that $\ker \Phi$ is trivial (\cref{surjectiveInjectiveProp}).

  Suppose $(h_1, h_2) \in \ker \Phi$, that is:
  \[
    \Phi(h_1, h_2) = h_1 h_2 = e
  \]
  Then $h_1 = h^{-1}_{2}$.
  Note that $h_1 \in H_1$ and $h^{-1}_{2} \in H_2$ but since they are equal, we must have $h_1 = h^{-1}_{2} \in H_1 \cap H_2$.
  By \textbf{i}, $H_1 \cap H_2 = \{e\}$ so $(h_1, h_2) = e$.
  Therefore, $\ker \Phi$ is trivial so $\Phi$ is injective.
  Thus $\Phi$ is a bijective homomorphism so $G \cong H_1 \times H_2$.
\end{proof}
\begin{lemma}
  If $H_1 \cap H_2 = \{e\}$ and $|H_1|, |H_2| < \infty$, then $|H_1 H_2| = |H_1||H_2|$.
\end{lemma}
\begin{proof}
  We need to show that when we multiply a pair of elements, we cannot get a repeated element.
  This is equivalent to showing that the map $\phi: H_1 \times H_2 \to H_1 H_2,\ (h_1, h_2) \mapsto h_1 h_2$ is bijective.
  This is clearly surjective by definition of $H_1H_2$.

  For injectivity, let $h_1, h_1' \in H_1$ and $h_2, h_2' \in H_2$ and suppose that:
  \begin{align*}
    \phi(h_1, h_2) &= \phi(h_1', h_2') \\
    \implies h_1 h_2 &= h_1' h_2' \\
    \implies (h_1')^{-1} h_1 &= h_2'h^{-1}_{2}
  \end{align*}
  But $(h_1')^{-1} h_1 \in H_1$ and $h_2h^{-1}_{2} \in H_2$ so both are in $H_1 \cap H_2 = \{e\}$.
  Thus $(h_1')^{-1} h_1 = e \implies h_1 = h_1'$ and $h_2' h^{-1}_{2} = e \implies h_2 = h_2'$, that is $(h_1, h_2) = (h_1', h_2')$.
  Therefore each element in $H_1 H_2$ corresponds to exactly one pair $(h_1, h_2) \in H_1 \times H_2$ so $|H_1H_2| = |H_1||H_2|$.
\end{proof}
\begin{corollary}
  In particular, if $|H_1||H_2| = |G| < \infty$, then $H_1 \cap H_2 = \{e\} \implies G = H_1 H_2$. \label{dptShortcut}
\end{corollary}
\begin{proof}
  We know that $H_1 \cap H_2 = \{e\} \implies |H_1||H_2| = |H_1H_2| = |G|$ and, since $H_1, H_2 \leq G$ we must have $H_1H_2 \subseteq G$.
  So $H_1H_2 \subseteq G$ and $|H_1H_2| = |G|$, thus $G = H_1 H_2$.
\end{proof}
\begin{remark}
  This shows us that if $|G| < \infty$ and $|G| = |H_1||H_2|$, then we need only show \textbf{i} and \textbf{ii} to apply the direct product theorem (\cref{directProductTheorem}).
\end{remark}
\begin{lemma}[Groups of Order 4]
  If $|G| = 4$, then either:
  \[
    G \cong C_4 \text{ or } G \cong K_4
  \]
\end{lemma}
\begin{proof}
  By Lagrange's Theorem (\cref{orderDivCorollary}), every nontrivial element of $G$ has order 2 or 4.

  If there is $g \in G$ such that $\ord(g) = 4$, then $G$ is generated by $g$ so $G \cong C_4$.

  Otherwise, every nontrivial element has order 2.
  Let $a, b \in G$ be distinct elements such that $\ord(a) = \ord(b) = 2$.
  Let $H_1 = \langle a \rangle$ and $H_2 = \langle b \rangle$, it is immediate that $H_1 \cap H_2 = \{e\}$ (\textbf{i} \tick), and since $|H_1||H_2| = 4 = |G|$ by \cref{dptShortcut}, $G = H_1H_2$ (\textbf{iii} \tick).

  Finally, since $\ord(ab) = 2$:
  \begin{align*}
    abab &= e \\
    \implies aba^{-1}b^{-1} &= e \text{ as $a, b$ self inverse} \\
    \implies ab &= ba
  \end{align*}
  So \textbf{ii} holds (See Sheet 1 Q11).
  Thus, $G \cong \langle a \rangle \times \langle b \rangle \cong C_2 \times C_2$.
\end{proof}
\section{Groups of Order 6}
Another nice application of the direct product theorem is to figure out when a product of cyclic groups is again cyclic.
\begin{theorem}[Chinese Remainder Theorem]
  If $\hcf(m, n) = 1$, then:
  \[
    C_m \times C_n \cong C_{nm}
  \]
\end{theorem}
\begin{proof}
  Let $C_{mn} = \langle g \rangle$, i.e. $o(g) = nm$.
  Set $H_1 = \langle g^{n} \rangle \leq C_{nm}$ and $H_2 = \langle g^{m} \rangle \leq C_{nm}$.

  $g^{nm} = e \implies o(g^{n}) \mid m \implies o(g^{n}) \leq m$.
  Suppose $o(g^{n}) = t < m$, then $g^{tn} = e$ but $tn < mn$ which contradicts $o(g) = nm$.
  Thus $o(g^{n}) = m$ and similarly $o(g^{m}) = n$, so $H_1 \cong C_m$ and $H_2 \cong C_n$.

  We will check the hypotheses of the direct product theorem.
  \begin{enumerate}
    \item Note that $g^{k} \in H_1 \iff n \mid k$ and $g^{k} \in H_2 \iff m \mid k$.
      Therefore,
      \[
        g^{k} \in H_1 \cap H_2 \iff m \mid k \text{ and } n \mid k \iff mn = \lcm(m, n) \mid k
      \]
      Therefore $g^{k} \in H_1 \cap H_2 \implies g^{k} = e$ so $H_1 \cap H_2 = \{e\}$.
    \item $C_{nm}$ is abelian so \textbf{ii} is immediate.
    \item $|C_n||C_m| = nm = |C_{nm}|$ so by \cref{dptShortcut}, \textbf{iii} holds.
  \end{enumerate}
  Therefore by \cref{directProductTheorem}, $C_{nm} \cong H_1 \times H_2 \cong C_m \times C_n$.
\end{proof}
\begin{lemma}[Groups of Order 6]
  If $|G| = 6$, then either: \label{groupsOfOrder6}
  \[
    G \cong C_6 \text{ or } G \cong D_6
  \]
\end{lemma}
\begin{proof}
  By Cauchy's Theorem (\cref{cauchysTheorem}), there are $r, s \in G$ such that $\ord(r) = 3$ and $\ord(s) = 2$.
  Since $|\langle r \rangle| = 3$, we have $|G : \langle r \rangle| = \frac{6}{3} = 2$.
  Furthermore, $s \notin \langle r \rangle$ since everything in $\langle r \rangle$ has order 3 or 1 and $\ord(s) = 2$.

  Since $|G : \langle r \rangle| = 2$, $\langle r \rangle$ has two cosets.
  One must be $\langle r \rangle$ itself and since cosets partition, the other coset must be $G$ with $\langle r \rangle$ removed, that is, $G \setminus \langle r \rangle$.
  Therefore since $s \notin \langle r \rangle$:
  \[
    s\langle r \rangle = G \setminus \langle r \rangle = \langle r \rangle s
  \]
  so $sr = r^{i}s$ for some $i = 0, 1, 2$.
  \begin{proofcases}
    \begin{case}{$i = 0$}
      Then $sr = s$ so $r = e$ which contradicts $\ord(r) = 3$ \cross
    \end{case}
    \begin{case}{$i = 1$}
      Then $sr = rs$, so $\langle s \rangle$ and $\langle r \rangle$ commute,
      $|\langle r \rangle| |\langle s \rangle| = 6 = |G|$ and $\langle r \rangle \cap \langle s \rangle = \{e\}$ so by \cref{directProductTheorem}:
      \[
        G \cong \langle r \rangle \times \langle s \rangle \cong C_3 \times C_2.
      \]
      Moreover, since $\hcf(3, 2) = 1$, by the Chinese remainder theorem, $C_3 \times C_2 \cong C_6$ so $G \cong C_6$ \tick
    \end{case}
    \begin{case}{$i = 2$}
      Then $sr = r^2s = r^{-1}s$, which is the dihedral relation.
      So by \cref{dihedralIsomorphic}, $G \cong D_6$ \tick
    \end{case}
  \end{proofcases}
\end{proof}
\begin{remark}
  $S_3$ is a non-abelian group of order 6, so $S_3 \cong D_6$.

  We can see this more directly without the lemma by labelling the vertices of the triangle 1, 2, 3 and then each permutation in $S_3$ simply permutes the vertices of the triangle.
  These permutations are always isometries since there are only three sides so adjacent vertices always remain adjacent.
\end{remark}
\section{Groups of Order 8}
We already know the following groups of order 8:
\[
  C_2 \times C_2 \times C_2,\ C_4 \times C_2,\ C_8,\ D_8
\]
\begin{remark}
  None of these examples are isomorphic to each other:
  \begin{itemize}
    \item $D_8$ is non-abelian so cannot be isomorphic to any of the others as they are all abelian.
    \item $C_8$ has an element of order 8 which none of the other groups can have otherwise they would have to be $C_8$.
    \item $C_2 \times C_2 \times C_2$ only has elements of order 2 but $C_4 \times C_2$ has an element of order 4 so they cannot be isomorphic.
  \end{itemize}
\end{remark}
\begin{example}[The Quaternion Group]
  Let $Q_8$:
  \[
    Q_8 = \left\{\begin{pmatrix}
    \pm 1 & 0 \\
    0 & \pm 1 \\
    \end{pmatrix},\
    \begin{pmatrix}
    \pm i & 0 \\
    0 & \mp i \\
    \end{pmatrix},\
    \begin{pmatrix}
    0 & \pm1 \\
    \mp 1 & 0 \\
    \end{pmatrix},\
    \begin{pmatrix}
    0 & \pm i \\
    \pm i & 0 \\
    \end{pmatrix}\right\}
  \]
  where $i^2 = -1$.

  It is easy to check that this is a group under matrix multiplication.
  Writing out these matrices is quite clunky so we usually use Hamilton's notation:
  \[
    1 = \begin{pmatrix}
    1 & 0 \\
    0 & 1 \\
    \end{pmatrix},\
    i = \begin{pmatrix}
    i & 0 \\
    0 & -i \\
    \end{pmatrix},\
    j = \begin{pmatrix}
    0 & 1 \\
    -1 & 0 \\
    \end{pmatrix},\
    k = \begin{pmatrix}
    0 & i \\
    i & 0 \\
    \end{pmatrix}
  \]
  So then the group consists of:
  \[
    Q_8 = \{\pm 1,\ \pm i,\ \pm j,\ \pm k\}
  \]
  \textbf{Properties of $Q_8$}
  \begin{itemize}
    \item $i^2 = j^2 = k^2 = -1$, so $\pm i, \pm j, \pm k$ all have order 4.
    \item $(-1)i = -i$, $(-1)j = -j$, $(-1)k = -k,\ldots$
    \item $ij = k$, $jk = i$, $ki = j$ and $ji = -k$, $kj = -i$, $ik = -j$ (can all be derived from $ijk = -1$)
    \item $-1$ commutes with everything.
    \item $-1$ is the only element of order 2, all other nontrivial elements have order 4.
    \item $Q_8 = \langle \{i, j\} \rangle$ or any other pair of $\{\pm i, \pm j, \pm k\}$.
  \end{itemize}
  This is the \textit{Quaternion Group}.
  Since $ji = -k \neq k = ij$, $Q_8$ is non-abelian so:
  \[
    Q_8 \centernot \cong C_2 \times C_2 \times C_2 \text{ or } C_4 \times C_2 \text{ or } C_8
  \]
  Note that $D_8$ has 5 elements of order 2 but $Q_8$ only has 1 element of order 2, namely $-1$, so $Q_8 \centernot\cong D_8$.
  Therefore $Q_8$ is not isomorphic to any of the earlier examples.
\end{example}
\begin{lemma}[Groups of Order 8]
  If $|G| = 8$, then $G$ is isomorphic to exactly one of:
  \begin{itemize}
    \item $C_2 \times C_2 \times C_2$
    \item $C_4 \times C_2$
    \item $C_8$
    \item $D_8$
    \item $Q_8$
  \end{itemize}
\end{lemma}
\begin{proof}
  We know that none of the groups in the above list are isomorphic to one another and so we just need to show that $G$ is always isomorphic to one of them.

  By Lagrange's Theorem (\cref{orderDivCorollary}), every element has order 1, 2, 4, or 8.

  \textbf{Element of Order 8 - }
  If there is an element of order 8, then $G \cong C_8$ \tick

  \textbf{All nontrivial elements order 2 -}
  Suppose that every nontrivial element has order 2.
  We can then choose 3 nontrivial elements $a, b, c$ such that $b \neq a$ and $c \neq a, b, ab$.

  Consider the subgroups $\langle a \rangle$, $\langle b \rangle$, and $\langle c \rangle$, because of how we chose $a$, $b$, and $c$, the pairwise intersections of these subgroups are trivial.
  We also know from Q11 Sheet 1 that if every nontrivial element has order 2, then the group must be abelian, thus, $\langle a \rangle$, $\langle b \rangle$ and $\langle c \rangle$ commute.
  So by direct product theorem (\cref{directProductTheorem}):
  \[
    \langle a \rangle\langle b \rangle \cong \langle a \rangle \times \langle b \rangle \cong C_2 \times C_2
  \]
  Moreover, as $|G| = |\langle a \rangle\langle b \rangle||\langle c \rangle|$ and $(\langle a \rangle\langle b \rangle) \cap \langle c \rangle = \{ e\}$, we can apply direct product theorem again to yield:
  \[
    G \cong (\langle a \rangle \langle b \rangle) \times \langle c \rangle \cong (\langle a \rangle \times \langle b \rangle) \times \langle c \rangle  \]
  Since the direct product is associative, $G \cong C_2 \times C_2 \times C_2$ \tick

  \textbf{All nontrivial elements order 2 or 4 -}
  Therefore, we may suppose that there is an $a \in G$ such that $\ord(a) = 4$.

  Let $b \in G \setminus \langle a \rangle$ such that $b \neq e$.
  Since $|\langle a \rangle| = 4$, by Lagrange's Theorem, $|G : \langle a \rangle| = 2$.
  Similarly to in the proof of \cref{groupsOfOrder6}, we can use the fact that there is only two cosets of $\langle a \rangle$ so both the left and right cosets with respect to $b$ must be equal.
  That is:
  \[
    b \langle a \rangle = G \setminus \langle a \rangle = \langle a \rangle b
  \]
  Firstly, note that that $b\langle a \rangle \cup \langle a \rangle = G$, and so $G = \langle \{a, b\} \rangle$.

  We also have $ba = a^{i}b$ for some $i \in \{0, 1, 2, 3\}$ (this is much more useful than trying to consider $b = a^{i}b$ as then we just get $i = 0$).
  \begin{proofcases}
    \begin{case}{$i = 0$}
      Then $ba = b$ so $a = e$, which is a contradiction as $\ord(a) = 4$ \cross
    \end{case}
    \begin{case}{$i = 1$}
      Then $ba = ab$ and so $G$ is abelian as it is generated by $\{a, b\}$.
    \end{case}
    \begin{case}{$i = 2$}
      Then $ba = a^2 b$ so $bab^{-1} = a^2$.
      $\ord(a) = 4$ so $\ord(a^2) = \ord(bab^{-1}) = 2$.
      But from Q1 Sheet 2, we know what the order of an elements conjugate is equal to the order of the element so we have arrived at a contradiction \cross
    \end{case}
    \begin{case}{$i = 3$}
      Then $ba = a^3b = a^{-1}b$, which is the dihedral relation.
    \end{case}
  \end{proofcases}
  So the only cases that can proceed are $i = 1, 3$.

  To proceed, it would be useful to know something about the order of $b$.
  Suppose that $b^2 \notin \langle a \rangle \implies b^2 \in b\langle a \rangle \implies b^2 = ba^{j}$ for some $j$, then $b = a^{j}$ so $b \in \langle a \rangle$ which is a contradiction.
  Therefore, we must have $b^2 \in \langle a \rangle$.
  Again, we get several cases:
  \begin{proofcases}
    \begin{case}{$b^2 = e$}
      Considering the valid cases $i = 1, 3$ from above:
      \begin{proofcases}
        \begin{case}{$i = 1$}
          So $G$ is abelian and $\ord(b) = 2$.
          Since $b \notin \langle a \rangle$, $\langle a \rangle \cap \langle b \rangle = \{e\}$.
          Further $|G| = 4 \cdot 2$ so, by direct product theorem, $G \cong C_4 \times C_2$ \tick
        \end{case}
        \begin{case}{$i = 3$}
          So $\ord(a) = 4$, $\ord(b) = 2$, $ba = a^{-1}b$, and $G = \langle \{a, b\} \rangle$
          Therefore, by \cref{dihedralIsomorphic}, $G \cong D_8$ \tick
        \end{case}
      \end{proofcases}
    \end{case}
    \begin{case}{$b^2 = a$}
      Since $\ord(a) = 4$, we must have $\ord(b) = 8$ which is a contradiction as we said all nontrivial elements have order 2 or 4.
    \end{case}
    \begin{case}{$b^2 = a^2$}
      Again considering the valid cases $i = 1, 3$ from above:
      \begin{proofcases}
        \begin{case}{$i = 1$}
          So $G$ is abelian and $b^2 = a^2$.
          Since $G$ is abelian, we should be aiming to show that it is isomorphic to $C_4 \times C_2$ so we need to find an element of order 2.
          Notice that:
          \[
            (ab)^2 = abab = a^2b^2 = a^4 = e
          \]
          Since $b \notin \langle a \rangle$, $ab \neq e$ and so $o(ab) = 2$.
          $ab \notin \langle a \rangle$ else $b \in \langle a \rangle$, so $\langle ab \rangle \cap \langle a \rangle = \{e\}$.
          As in \textbf{1.i}, by the direct product theorem, $G \cong C_4 \times C_2$ \tick
        \end{case}
        \begin{case}{$i = 3$}
          So $a^4 = e$, $b^2 = a^2$, $ba = a^{-1}b$, and $G = \langle \{a, b\} \rangle$.
          Since $Q_8 = \langle \{i, j\} \rangle$, we can show that the homomorphism $\phi: Q_8 \to G$ defined by $i \mapsto a,\ j \mapsto b$ is well defined.
          Notice first that since $\phi$ is a homomorphism $\phi(i^2) = \phi(-1) = a^2$ and $\phi(ij) = \phi(k) = ab$.
          We then check that:
          \begin{align*}
            \phi(ji) &= \phi(j)\phi(i) = ba = a^{-1}b = a^2 ab = \phi(-1)\phi(k) = \phi(-k) \\
            \phi(jk) &= \phi(j)\phi(k) = bab = a^{-1}b^2 = a^{5} = a = \phi(i) \\
            \phi(kj) &= \phi(k)\phi(j) = ab^2 = \phi(i)\phi(-1) = \phi(-i) \\
            \phi(ik) &= \phi(i)\phi(k) = a^2b = \phi(-1)\phi(j) = \phi(-j) \\
            \phi(ki) &= \phi(k)\phi(i) = aba = a a^{-1}b = b = \phi(j)
          \end{align*}
          so $\phi$ is a well defined homomorphism.

          Since $G = \langle \{a, b\} \rangle$, $G \leq \im \phi \leq G \implies \im \phi = G$ and so $\phi$ is a surjective homomorphism between two groups of equal size.
          Thus $G \cong Q_8$.
        \end{case}
      \end{proofcases}
    \end{case}
    \begin{case}{$b^2 = a^3$}
      So $b^2 = a^{-1}$, $\ord(a^{-1}) = \ord(a) = 4$, therefore $\ord(b) = 8$ which is a contradiction.
    \end{case}
  \end{proofcases}
  So in all valid cases $G$ is isomorphic to exactly one of the groups in our list.
\end{proof}
